12 // DB contents are stored in a set of blocks, each of which holds a
13 // sequence of key,value pairs. Each block may be compressed before
14 // being stored in a file. The following enum describes which
15 // compression method (if any) is used to compress a block.
16 
17 enum CompressionType : unsigned char {
18  // NOTE: do not change the values of existing entries, as these are

21  kSnappyCompression = 0x1,
22  kZlibCompression = 0x2,
23  kBZip2Compression = 0x3,
24  kLZ4Compression = 0x4,

26  kXpressCompression = 0x6,
27  kZSTD = 0x7,
28 
29  // Only use kZSTDNotFinalCompression if you have to use ZSTD lib older than
30  // 0.8.0 or consider a possibility of downgrading the service or copying
31  // the database files to another service running with an older version of
32  // RocksDB that doesn't have kZSTD. Otherwise, you should use kZSTD. We will
33  // eventually remove the option from the public API.
34  kZSTDNotFinalCompression = 0x40,
