29 // A FullFilterBlockBuilder is used to construct a full filter for a
30 // particular Table. It generates a single string which is stored as
31 // a special block in the Table.
32 // The format of full filter block is:
33 // +----------------------------------------------------------------+
34 // | full filter for all keys in sst file |
35 // +----------------------------------------------------------------+
36 // The full filter can be very large. At the end of it, we put
37 // num_probes: how many hash functions are used in bloom filter
38 //
39 class FullFilterBlockBuilder : public FilterBlockBuilder {
40  public:
41  explicit FullFilterBlockBuilder(const SliceTransform* prefix_extractor,
42  bool whole_key_filtering,
43  FilterBitsBuilder* filter_bits_builder);
44  // No copying allowed
45  FullFilterBlockBuilder(const FullFilterBlockBuilder&) = delete;
46  void operator=(const FullFilterBlockBuilder&) = delete;
47 
48  // bits_builder is created in filter_policy, it should be passed in here
49  // directly. and be deleted here
50  ~FullFilterBlockBuilder() {}
51 
52  virtual bool IsBlockBased() override { return false; }
53  virtual void StartBlock(uint64_t /*block_offset*/) override {}
54  virtual void Add(const Slice& key_without_ts) override;
55  virtual bool IsEmpty() const override { return !any_added_; }
56  virtual size_t EstimateEntriesAdded() override;
57  virtual Slice Finish(const BlockHandle& tmp, Status* status) override;
58  using FilterBlockBuilder::Finish;
59 
60  protected:
61  virtual void AddKey(const Slice& key);
62  std::unique_ptr<FilterBitsBuilder> filter_bits_builder_;
63  virtual void Reset();
64  void AddPrefix(const Slice& key);
65  const SliceTransform* prefix_extractor() { return prefix_extractor_; }
66  const std::string& last_prefix_str() const { return last_prefix_str_; }
67 
68  private:
69  // important: all of these might point to invalid addresses
70  // at the time of destruction of this filter block. destructor
71  // should NOT dereference them.
72  const SliceTransform* prefix_extractor_;
73  bool whole_key_filtering_;
74  bool last_whole_key_recorded_;
75  std::string last_whole_key_str_;
76  bool last_prefix_recorded_;
77  std::string last_prefix_str_;
78  // Whether prefix_extractor_->InDomain(last_whole_key_) is true.
79  // Used in partitioned filters so that the last prefix from the previous
80  // filter partition will be added to the current partition if
81  // last_key_in_domain_ is true, regardless of the current key.
82  bool last_key_in_domain_;
83  bool any_added_;
84  std::unique_ptr<const char[]> filter_data_;
85 };
