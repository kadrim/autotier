19 // A "skipped" test has a specific meaning in Facebook infrastructure: the
20 // test is in good shape and should be run, but something about the
21 // compilation or execution environment means the test cannot be run.
22 // Specifically, there is a hole in intended testing if any
23 // parameterization of a test (e.g. Foo/FooTest.Bar/42) is skipped for all
24 // tested build configurations/platforms/etc.
25 //
26 // If GTEST_SKIP is available, use it. Otherwise, define skip as success.
27 //
28 // The GTEST macros do not seem to print the message, even with -verbose,
29 // so these print to stderr. Note that these do not exit the test themselves;
30 // calling code should 'return' or similar from the test.
31 #ifdef GTEST_SKIP_
32 #define ROCKSDB_GTEST_SKIP(m) \
33  do { \
34  fputs("SKIPPED: " m "\n", stderr); \
35  GTEST_SKIP_(m); \
36  } while (false) /* user ; */
37 #else
38 #define ROCKSDB_GTEST_SKIP(m) \
39  do { \
40  fputs("SKIPPED: " m "\n", stderr); \
41  GTEST_SUCCESS_("SKIPPED: " m); \
42  } while (false) /* user ; */
43 #endif
44 
45 // We add "bypass" as an alternative to ROCKSDB_GTEST_SKIP that is allowed to
46 // be a permanent condition, e.g. for intentionally omitting or disabling some
47 // parameterizations for some tests. (Use _DISABLED at the end of the test
48 // name to disable an entire test.)
49 #define ROCKSDB_GTEST_BYPASS(m) \
50  do { \
51  fputs("BYPASSED: " m "\n", stderr); \
52  GTEST_SUCCESS_("BYPASSED: " m); \
53  } while (false) /* user ; */
