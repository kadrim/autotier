43 struct ComparatorJniCallbackOptions {
44 
45  // Set the synchronisation type used to guard the reused buffers.
46  // Only used if max_reused_buffer_size > 0.
47  // Default: ADAPTIVE_MUTEX
48  ReusedSynchronisationType reused_synchronisation_type =
49  ReusedSynchronisationType::ADAPTIVE_MUTEX;
50 
51  // Indicates if a direct byte buffer (i.e. outside of the normal
52  // garbage-collected heap) is used for the callbacks to Java,
53  // as opposed to a non-direct byte buffer which is a wrapper around
54  // an on-heap byte[].
55  // Default: true
56  bool direct_buffer = true;
57 
58  // Maximum size of a buffer (in bytes) that will be reused.
59  // Comparators will use 5 of these buffers,
60  // so the retained memory size will be 5 * max_reused_buffer_size.
61  // When a buffer is needed for transferring data to a callback,
62  // if it requires less than max_reused_buffer_size, then an
63  // existing buffer will be reused, else a new buffer will be
64  // allocated just for that callback. -1 to disable.
65  // Default: 64 bytes
66  int32_t max_reused_buffer_size = 64;
67 };
