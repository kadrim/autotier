27 /*
28  class Endpoint allows to define prefix ranges.
29 
30  Prefix ranges are introduced below.
31 
32  == Basic Ranges ==
33  Let's start from basic ranges. Key Comparator defines ordering of rowkeys.
34  Then, one can specify finite closed ranges by just providing rowkeys of their
35  endpoints:
36 
37  lower_endpoint <= X <= upper_endpoint
38 
39  However our goal is to provide a richer set of endpoints. Read on.
40 
41  == Lexicographic ordering ==
42  A lexicographic (or dictionary) ordering satisfies these criteria: If there
43  are two keys in form
44  key_a = {prefix_a, suffix_a}
45  key_b = {prefix_b, suffix_b}
46  and
47  prefix_a < prefix_b
48  then
49  key_a < key_b.
50 
51  == Prefix ranges ==
52  With lexicographic ordering, one may want to define ranges in form
53 
54  "prefix is $PREFIX"
55 
56  which translates to a range in form
57 
58  {$PREFIX, -infinity} < X < {$PREFIX, +infinity}
59 
60  where -infinity will compare less than any possible suffix, and +infinity
61  will compare as greater than any possible suffix.
62 
63  class Endpoint allows to define these kind of rangtes.
64 
65  == Notes ==
66  BytewiseComparator and ReverseBytewiseComparator produce lexicographic
67  ordering.
68 
69  The row comparison function is able to compare key prefixes. If the data
70  domain includes keys A and B, then the comparison function is able to compare
71  equal-length prefixes:
72 
73  min_len= min(byte_length(A), byte_length(B));
74  cmp(Slice(A, min_len), Slice(B, min_len)); // this call is valid
75 
76  == Other options ==
77  As far as MyRocks is concerned, the alternative to prefix ranges would be to
78  support both open (non-inclusive) and closed (inclusive) range endpoints.
79 */
80

138  // If a transaction has a snapshot set, the transaction will ensure that
139  // any keys successfully written(or fetched via GetForUpdate()) have not
140  // been modified outside of this transaction since the time the snapshot was
141  // set.
142  // If a snapshot has not been set, the transaction guarantees that keys have
143  // not been modified since the time each key was first written (or fetched via
144  // GetForUpdate()).
145  //
146  // Using SetSnapshot() will provide stricter isolation guarantees at the
147  // expense of potentially more transaction failures due to conflicts with
148  // other writes.
149  //
150  // Calling SetSnapshot() has no effect on keys written before this function
151  // has been called.
152  //
153  // SetSnapshot() may be called multiple times if you would like to change
154  // the snapshot used for different operations in this transaction.
155  //
156  // Calling SetSnapshot will not affect the version of Data returned by Get()
157  // methods. See Transaction::Get() for more details.
158  virtual void SetSnapshot() = 0;
159 
160  // Similar to SetSnapshot(), but will not change the current snapshot
161  // until Put/Merge/Delete/GetForUpdate/MultigetForUpdate is called.
162  // By calling this function, the transaction will essentially call
163  // SetSnapshot() for you right before performing the next write/GetForUpdate.
164  //
165  // Calling SetSnapshotOnNextOperation() will not affect what snapshot is
166  // returned by GetSnapshot() until the next write/GetForUpdate is executed.
167  //
168  // When the snapshot is created the notifier's SnapshotCreated method will
169  // be called so that the caller can get access to the snapshot.
170  //
171  // This is an optimization to reduce the likelihood of conflicts that
172  // could occur in between the time SetSnapshot() is called and the first
173  // write/GetForUpdate operation. Eg, this prevents the following
174  // race-condition:
175  //
176  // txn1->SetSnapshot();
177  // txn2->Put("A", ...);
178  // txn2->Commit();
179  // txn1->GetForUpdate(opts, "A", ...); // FAIL!
180  virtual void SetSnapshotOnNextOperation(
181  std::shared_ptr<TransactionNotifier> notifier = nullptr) = 0;
182 
183  // Returns the Snapshot created by the last call to SetSnapshot().
184  //
185  // REQUIRED: The returned Snapshot is only valid up until the next time
186  // SetSnapshot()/SetSnapshotOnNextSavePoint() is called, ClearSnapshot()
187  // is called, or the Transaction is deleted.

190  // Clears the current snapshot (i.e. no snapshot will be 'set')

195  // Calling ClearSnapshot() has no effect on keys written before this function

200  virtual void ClearSnapshot() = 0;

304  // Read this key and ensure that this transaction will only
305  // be able to be committed if this key is not written outside this
306  // transaction after it has first been read (or after the snapshot if a
307  // snapshot is set in this transaction and do_validate is true). If
308  // do_validate is false, ReadOptions::snapshot is expected to be nullptr so
309  // that GetForUpdate returns the latest committed value. The transaction
310  // behavior is the same regardless of whether the key exists or not.
311  //
312  // Note: Currently, this function will return Status::MergeInProgress
313  // if the most recent write to the queried key in this batch is a Merge.
314  //
315  // The values returned by this function are similar to Transaction::Get().
316  // If value==nullptr, then this function will not read any data, but will
317  // still ensure that this key cannot be written to by outside of this
318  // transaction.
319  //
320  // If this transaction was created by an OptimisticTransaction, GetForUpdate()
321  // could cause commit() to fail. Otherwise, it could return any error
322  // that could be returned by DB::Get().
323  //
324  // If this transaction was created by a TransactionDB, it can return
325  // Status::OK() on success,
326  // Status::Busy() if there is a write conflict,
327  // Status::TimedOut() if a lock could not be acquired,
328  // Status::TryAgain() if the memtable history size is not large enough
329  // (See max_write_buffer_size_to_maintain)
330  // Status::MergeInProgress() if merge operations cannot be resolved.
331  // or other errors if this key could not be read.

333  ColumnFamilyHandle* column_family,
334  const Slice& key, std::string* value,
335  bool exclusive = true,
336  const bool do_validate = true) = 0;

341  ColumnFamilyHandle* column_family,

343  bool exclusive = true,
344  const bool do_validate = true) {
345  if (pinnable_val == nullptr) {
346  std::string* null_str = nullptr;
347  return GetForUpdate(options, column_family, key, null_str, exclusive,
348  do_validate);
349  } else {
350  auto s = GetForUpdate(options, column_family, key,
351  pinnable_val->GetSelf(), exclusive, do_validate);
352  pinnable_val->PinSelf();
353  return s;
354  }
355  }
