105  // share_files_with_checksum supports table and blob files.
106  //

140  // ShareFilesNaming describes possible naming schemes for backup
141  // table and blob file names when they are stored in the
142  // shared_checksum directory (i.e., both share_table_files and
143  // share_files_with_checksum are true).
144  enum ShareFilesNaming : uint32_t {
145  // Backup blob filenames are <file_number>_<crc32c>_<file_size>.blob and
146  // backup SST filenames are <file_number>_<crc32c>_<file_size>.sst
147  // where <crc32c> is an unsigned decimal integer. This is the
148  // original/legacy naming scheme for share_files_with_checksum,
149  // with two problems:
150  // * At massive scale, collisions on this triple with different file
151  // contents is plausible.
152  // * Determining the name to use requires computing the checksum,
153  // so generally requires reading the whole file even if the file
154  // is already backed up.
155  //
156  // ** ONLY RECOMMENDED FOR PRESERVING OLD BEHAVIOR **
157  kLegacyCrc32cAndFileSize = 1U,
158 
159  // Backup SST filenames are <file_number>_s<db_session_id>.sst. This
160  // pair of values should be very strongly unique for a given SST file
161  // and easily determined before computing a checksum. The 's' indicates
162  // the value is a DB session id, not a checksum.
163  //
164  // Exceptions:
165  // * For blob files, kLegacyCrc32cAndFileSize is used as currently
166  // db_session_id is not supported by the blob file format.
167  // * For old SST files without a DB session id, kLegacyCrc32cAndFileSize
168  // will be used instead, matching the names assigned by RocksDB versions
169  // not supporting the newer naming scheme.
170  // * See also flags below.
171  kUseDbSessionId = 2U,
172 
173  kMaskNoNamingFlags = 0xffffU,
174 
175  // If not already part of the naming scheme, insert
176  // _<file_size>
177  // before .sst and .blob in the name. In case of user code actually parsing
178  // the last _<whatever> before the .sst and .blob as the file size, this
179  // preserves that feature of kLegacyCrc32cAndFileSize. In other words, this
180  // option makes official that unofficial feature of the backup metadata.
181  //
182  // We do not consider SST and blob file sizes to have sufficient entropy to
183  // contribute significantly to naming uniqueness.
184  kFlagIncludeFileSize = 1U << 31,
185 
186  kMaskNamingFlags = ~kMaskNoNamingFlags,
187  };
188 
189  // Naming option for share_files_with_checksum table and blob files. See
190  // ShareFilesNaming for details.
191  //
192  // Modifying this option cannot introduce a downgrade compatibility issue
193  // because RocksDB can read, restore, and delete backups using different file
194  // names, and it's OK for a backup directory to use a mixture of table and
195  // blob files naming schemes.
196  //
197  // However, modifying this option and saving more backups to the same
198  // directory can lead to the same file getting saved again to that
199  // directory, under the new shared name in addition to the old shared
200  // name.
201  //
202  // Default: kUseDbSessionId | kFlagIncludeFileSize
203  //
204  // Note: This option comes into effect only if both share_files_with_checksum
205  // and share_table_files are true.
206  ShareFilesNaming share_files_with_checksum_naming;

239 inline BackupEngineOptions::ShareFilesNaming operator&(
240  BackupEngineOptions::ShareFilesNaming lhs,
241  BackupEngineOptions::ShareFilesNaming rhs) {
242  uint32_t l = static_cast<uint32_t>(lhs);
243  uint32_t r = static_cast<uint32_t>(rhs);
244  assert(r == BackupEngineOptions::kMaskNoNamingFlags ||
245  (r & BackupEngineOptions::kMaskNoNamingFlags) == 0);
246  return static_cast<BackupEngineOptions::ShareFilesNaming>(l & r);
247 }
248 
249 inline BackupEngineOptions::ShareFilesNaming operator|(
250  BackupEngineOptions::ShareFilesNaming lhs,
251  BackupEngineOptions::ShareFilesNaming rhs) {
252  uint32_t l = static_cast<uint32_t>(lhs);
253  uint32_t r = static_cast<uint32_t>(rhs);
254  assert((r & BackupEngineOptions::kMaskNoNamingFlags) == 0);
255  return static_cast<BackupEngineOptions::ShareFilesNaming>(l | r);
256 }

519 // A backup engine for organizing and managing backups.
520 // This class is not user-extensible.
521 //
522 // This class declaration adds "Write" operations in addition to the
523 // operations from BackupEngineAppendOnlyBase and BackupEngineReadOnlyBase.
524 //
525 // # Concurrency between threads on the same BackupEngine* object
526 //
527 // As of version 6.20, BackupEngine* operations are generally thread-safe,
528 // using a read-write lock, though single-thread operation is still
529 // recommended to avoid TOCTOU bugs. Specifically, particular kinds of
530 // concurrent operations behave like this:
531 //
532 // op1\op2| Read | Append | Write
533 // -------|-------|--------|--------
534 // Read | conc | block | block
535 // Append | block | block | block
536 // Write | block | block | block
537 //
538 // conc = operations safely proceed concurrently
539 // block = one of the operations safely blocks until the other completes.
540 // There is generally no guarantee as to which completes first.
541 //
542 // StopBackup is the only operation that affects an ongoing operation.
543 //
544 // # Interleaving operations between BackupEngine* objects open on the
545 // same backup_dir
546 //
547 // It is recommended only to have one BackupEngine* object open for a given
548 // backup_dir, but it is possible to mix / interleave some operations
549 // (regardless of whether they are concurrent) with these caveats:
550 //
551 // op1\op2| Open | Read | Append | Write
552 // -------|--------|--------|--------|--------
553 // Open | conc | conc | atomic | unspec
554 // Read | conc | conc | old | unspec
555 // Append | atomic | old | unspec | unspec
556 // Write | unspec | unspec | unspec | unspec
557 //
558 // Special case: Open with destroy_old_data=true is really a Write
559 //
560 // conc = operations safely proceed, concurrently when applicable
561 // atomic = operations are effectively atomic; if a concurrent Append
562 // operation has not completed at some key point during Open, the
563 // opened BackupEngine* will never see the result of the Append op.
564 // old = Read operations do not include any state changes from other
565 // BackupEngine* objects; they return the state at their Open time.
566 // unspec = Behavior is unspecified, including possibly trashing the
567 // backup_dir, but is "memory safe" (no C++ undefined behavior)
568 //
569 class BackupEngine : public BackupEngineReadOnlyBase,
570  public BackupEngineAppendOnlyBase {
571  public:
572  virtual ~BackupEngine() {}
573 
574  // BackupEngineOptions have to be the same as the ones used in previous
