32 // NOTE that: this class is EXPERIMENTAL! May be changed in the future!
33 // This is an abstract class to write or transfer the data that is created by
34 // CacheDumper. We pack one block with its block type, dump time, block key in
35 // the block cache, block len, block crc32c checksum and block itself as a unit
36 // and it is stored via WritePacket. Before we call WritePacket, we must call
37 // WriteMetadata once, which stores the sequence number, block unit checksum,
38 // and block unit size.
39 // We provide file based CacheDumpWriter to store the metadata and its package
40 // sequentially in a file as the defualt implementation. Users can implement
41 // their own CacheDumpWriter to store/transfer the data. For example, user can
42 // create a subclass which transfer the metadata and package on the fly.
43 class CacheDumpWriter {
44  public:
45  virtual ~CacheDumpWriter() = default;
46 
47  // Called ONCE before the calls to WritePacket
48  virtual IOStatus WriteMetadata(const Slice& metadata) = 0;
49  virtual IOStatus WritePacket(const Slice& data) = 0;
50  virtual IOStatus Close() = 0;
51 };

53 // NOTE that: this class is EXPERIMENTAL! May be changed in the future!
54 // This is an abstract class to read or receive the data that is stored
55 // or transfered by CacheDumpWriter. Note that, ReadMetadata must be called
56 // once before we call a ReadPacket.
57 class CacheDumpReader {
58  public:
59  virtual ~CacheDumpReader() = default;
60  // Called ONCE before the calls to ReadPacket
61  virtual IOStatus ReadMetadata(std::string* metadata) = 0;
62  // Sets data to empty string on EOF
63  virtual IOStatus ReadPacket(std::string* data) = 0;
64  // (Close not needed)
65 };

73 // NOTE that: this class is EXPERIMENTAL! May be changed in the future!
74 // This the class to dump out the block in the block cache, store/transfer them
75 // via CacheDumpWriter. In order to dump out the blocks belonging to a certain
76 // DB or a list of DB (block cache can be shared by many DB), user needs to call
77 // SetDumpFilter to specify a list of DB to filter out the blocks that do not
78 // belong to those DB.
79 // A typical use case is: when we migrate a DB instance from host A to host B.
80 // We need to reopen the DB at host B after all the files are copied to host B.
81 // At this moment, the block cache at host B does not have any block from this
82 // migrated DB. Therefore, the read performance can be low due to cache warm up.
83 // By using CacheDumper before we shut down the DB at host A and using
84 // CacheDumpedLoader at host B before we reopen the DB, we can warmup the cache
85 // ahead. This function can be used in other use cases also.
86 class CacheDumper {
87  public:
88  virtual ~CacheDumper() = default;
89  // Only dump the blocks in the block cache that belong to the DBs in this list
90  virtual Status SetDumpFilter(std::vector<DB*> db_list) {
91  (void)db_list;
92  return Status::NotSupported("SetDumpFilter is not supported");
93  }
94  // The main function to dump out all the blocks that satisfy the filter
95  // condition from block cache to a certain CacheDumpWriter in one shot. This
96  // process may take some time.
97  virtual IOStatus DumpCacheEntriesToWriter() {
98  return IOStatus::NotSupported("DumpCacheEntriesToWriter is not supported");
99  }
100 };
