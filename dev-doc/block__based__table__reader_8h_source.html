47 // Reader class for BlockBasedTable format.
48 // For the format of BlockBasedTable refer to
49 // https://github.com/facebook/rocksdb/wiki/Rocksdb-BlockBasedTable-Format.
50 // This is the default table type. Data is chucked into fixed size blocks and
51 // each block in-turn stores entries. When storing data, we can compress and/or
52 // encode data efficiently within a block, which often results in a much smaller
53 // data size compared with the raw data size. As for the record retrieval, we'll
54 // first locate the block where target record may reside, then read the block to
55 // memory, and finally search that record within the block. Of course, to avoid
56 // frequent reads of the same block, we introduced the block cache to keep the
57 // loaded blocks in the memory.

469  // Generate a cache key prefix from the file
470  template <typename TCache, typename TFile>
471  static void GenerateCachePrefix(TCache* cc, TFile* file, char* buffer,
472  size_t* size,
473  const std::string& db_session_id,
474  uint64_t cur_file_num) {
475  // generate an id from the file
476  *size = file->GetUniqueId(buffer, kMaxCacheKeyPrefixSize);
477 
478  // If the prefix wasn't generated or was too long,
479  // create one based on the DbSessionId and curent file number if they
480  // are set. Otherwise, created from NewId()
481  if (cc != nullptr && *size == 0) {
482  if (db_session_id.size() == 20) {
483  // db_session_id is 20 bytes as defined.
484  memcpy(buffer, db_session_id.c_str(), 20);
485  char* end;
486  if (cur_file_num != 0) {
487  end = EncodeVarint64(buffer + 20, cur_file_num);
488  } else {
489  end = EncodeVarint64(buffer + 20, cc->NewId());
490  }
491  // kMaxVarint64Length is 10 therefore, the prefix is at most 30 bytes.
492  *size = static_cast<size_t>(end - buffer);
493  } else {
494  char* end = EncodeVarint64(buffer, cc->NewId());
495  *size = static_cast<size_t>(end - buffer);
496  }
497  }
498  }

696  // Override overflow() to handle `sputc()`. There are cases that will not go
697  // through `xsputn()` e.g. `std::endl` or an unsigned long long is written by
698  // `os.put()` directly and will call `sputc()` By internal implementation:
699  // int_type __CLR_OR_THIS_CALL sputc(_Elem _Ch) { // put a character
700  // return 0 < _Pnavail() ? _Traits::to_int_type(*_Pninc() = _Ch) :
701  // overflow(_Traits::to_int_type(_Ch));
702  // }
703  // As we explicitly disabled buffering (_Pnavail() is always 0), every write,
704  // not captured by xsputn(), becomes an overflow here.
705  int overflow(int ch = EOF) override {
706  if (ch != EOF) {
707  Status s = file_->Append(Slice((char*)&ch, 1));
708  if (s.ok()) {
709  return ch;
710  }
711  }
712  return EOF;
713  }
714 
715  std::streamsize xsputn(char const* p, std::streamsize n) override {
716  Status s = file_->Append(Slice(p, n));
717  if (!s.ok()) {
718  return 0;
719  }
720  return n;
721  }
722 
723  private:
724  WritableFile* file_;
725 };
