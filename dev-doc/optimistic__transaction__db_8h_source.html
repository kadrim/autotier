34 enum class OccValidationPolicy {
35  // Validate serially at commit stage, AFTER entering the write-group.
36  // Isolation validation is processed single-threaded(since in the
37  // write-group).
38  // May suffer from high mutex contention, as per this link:
39  // https://github.com/facebook/rocksdb/issues/4402
40  kValidateSerial = 0,
41  // Validate parallelly before commit stage, BEFORE entering the write-group to
42  // reduce mutex contention. Each txn acquires locks for its write-set
43  // records in some well-defined order.
44  kValidateParallel = 1
45 };

50  // works only if validate_policy == OccValidationPolicy::kValidateParallel
