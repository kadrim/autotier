163 // TableBuilder provides the interface used to build a Table
164 // (an immutable and sorted map from keys to values).
165 //
166 // Multiple threads can invoke const methods on a TableBuilder without
167 // external synchronization, but if any of the threads may call a
168 // non-const method, all threads accessing the same TableBuilder must use
169 // external synchronization.
170 class TableBuilder {
171  public:
172  // REQUIRES: Either Finish() or Abandon() has been called.
173  virtual ~TableBuilder() {}
174 
175  // Add key,value to the table being constructed.
176  // REQUIRES: key is after any previously added key according to comparator.
177  // REQUIRES: Finish(), Abandon() have not been called
178  virtual void Add(const Slice& key, const Slice& value) = 0;
179 
180  // Return non-ok iff some error has been detected.
181  virtual Status status() const = 0;
182 
183  // Return non-ok iff some error happens during IO.
184  virtual IOStatus io_status() const = 0;
185 
186  // Finish building the table.
187  // REQUIRES: Finish(), Abandon() have not been called
188  virtual Status Finish() = 0;
189 
190  // Indicate that the contents of this builder should be abandoned.
191  // If the caller is not going to call Finish(), it must call Abandon()
192  // before destroying this builder.
193  // REQUIRES: Finish(), Abandon() have not been called
194  virtual void Abandon() = 0;
195 
196  // Number of calls to Add() so far.
197  virtual uint64_t NumEntries() const = 0;
198 
199  // Whether the output file is completely empty. It has neither entries
200  // or tombstones.
201  virtual bool IsEmpty() const {
202  return NumEntries() == 0 && GetTableProperties().num_range_deletions == 0;
203  }
204 
205  // Size of the file generated so far. If invoked after a successful
206  // Finish() call, returns the size of the final generated file.
207  virtual uint64_t FileSize() const = 0;
208 
209  // Estimated size of the file generated so far. This is used when
210  // FileSize() cannot estimate final SST size, e.g. parallel compression
211  // is enabled.
212  virtual uint64_t EstimatedFileSize() const { return FileSize(); }
213 
214  // If the user defined table properties collector suggest the file to
215  // be further compacted.
216  virtual bool NeedCompact() const { return false; }
217 
218  // Returns table properties
219  virtual TableProperties GetTableProperties() const = 0;
220 
221  // Return file checksum
222  virtual std::string GetFileChecksum() const = 0;
223 
224  // Return file checksum function name
225  virtual const char* GetFileChecksumFuncName() const = 0;
226 };
