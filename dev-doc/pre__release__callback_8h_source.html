13 class PreReleaseCallback {
14  public:
15  virtual ~PreReleaseCallback() {}
16 
17  // Will be called while on the write thread after the write to the WAL and
18  // before the write to memtable. This is useful if any operation needs to be
19  // done before the write gets visible to the readers, or if we want to reduce
20  // the overhead of locking by updating something sequentially while we are on
21  // the write thread. If the callback fails, this function returns a non-OK
22  // status, the sequence number will not be released, and same status will be
23  // propagated to all the writers in the write group.
24  // seq is the sequence number that is used for this write and will be
25  // released.
26  // is_mem_disabled is currently used for debugging purposes to assert that
27  // the callback is done from the right write queue.
28  // If non-zero, log_number indicates the WAL log to which we wrote.
29  // index >= 0 specifies the order of callback in the same write thread.
30  // total > index specifies the total number of callbacks in the same write
31  // thread. Together with index, could be used to reduce the redundant
32  // operations among the callbacks.
33  virtual Status Callback(SequenceNumber seq, bool is_mem_disabled,
34  uint64_t log_number, size_t index, size_t total) = 0;
35 };
