20 // A handle for lookup result. The handle may not be immediately ready or
21 // have a valid value. The caller must call isReady() to determine if its
22 // ready, and call Wait() in order to block until it becomes ready.
23 // The caller must call value() after it becomes ready to determine if the
24 // handle successfullly read the item.
25 class SecondaryCacheResultHandle {
26  public:
27  virtual ~SecondaryCacheResultHandle() {}
28 
29  // Returns whether the handle is ready or not
30  virtual bool IsReady() = 0;
31 
32  // Block until handle becomes ready
33  virtual void Wait() = 0;
34 
35  // Return the value. If nullptr, it means the lookup was unsuccessful
36  virtual void* Value() = 0;
37 
38  // Return the size of value
39  virtual size_t Size() = 0;
40 };

44 // Cache interface for caching blocks on a secondary tier (which can include
45 // non-volatile media, or alternate forms of caching such as compressed data)
46 class SecondaryCache : public Customizable {
47  public:

50  static const char* Type() { return "SecondaryCache"; }

55  // Insert the given value into this cache. The value is not written
56  // directly. Rather, the SaveToCallback provided by helper_cb will be
57  // used to extract the persistable data in value, which will be written
58  // to this tier. The implementation may or may not write it to cache
59  // depending on the admission control policy, even if the return status is
60  // success.
61  virtual Status Insert(const Slice& key, void* value,
62  const Cache::CacheItemHelper* helper) = 0;
63 
64  // Lookup the data for the given key in this cache. The create_cb
65  // will be used to create the object. The handle returned may not be
66  // ready yet, unless wait=true, in which case Lookup() will block until
67  // the handle is ready
68  virtual std::unique_ptr<SecondaryCacheResultHandle> Lookup(
69  const Slice& key, const Cache::CreateCallback& create_cb, bool wait) = 0;
70 
71  // At the discretion of the implementation, erase the data associated
72  // with key
73  virtual void Erase(const Slice& key) = 0;
74 
75  // Wait for a collection of handles to become ready
76  virtual void WaitAll(std::vector<SecondaryCacheResultHandle*> handles) = 0;

78  virtual std::string GetPrintableOptions() const = 0;
