19 // Helper class that locks a mutex on construction and unlocks the mutex when
20 // the destructor of the MutexLock object is invoked.
21 //
22 // Typical usage:
23 //
24 // void MyClass::MyMethod() {
25 // MutexLock l(&mu_); // mu_ is an instance variable
26 // ... some complex code, possibly with multiple return paths ...
27 // }

29 class MutexLock {
30  public:
31  explicit MutexLock(port::Mutex *mu) : mu_(mu) {
32  this->mu_->Lock();
33  }
34  // No copying allowed
35  MutexLock(const MutexLock &) = delete;
36  void operator=(const MutexLock &) = delete;
37 
38  ~MutexLock() { this->mu_->Unlock(); }
39 
40  private:
41  port::Mutex *const mu_;
42 };

100 //
101 // SpinMutex has very low overhead for low-contention cases. Method names
102 // are chosen so you can use std::unique_lock or std::lock_guard with it.
103 //
104 class SpinMutex {
105  public:
106  SpinMutex() : locked_(false) {}
107 
108  bool try_lock() {
109  auto currently_locked = locked_.load(std::memory_order_relaxed);
110  return !currently_locked &&
111  locked_.compare_exchange_weak(currently_locked, true,
112  std::memory_order_acquire,
113  std::memory_order_relaxed);
114  }
115 
116  void lock() {
117  for (size_t tries = 0;; ++tries) {
118  if (try_lock()) {
119  // success
120  break;
121  }
122  port::AsmVolatilePause();
123  if (tries > 100) {
124  std::this_thread::yield();
125  }
126  }
127  }
128 
129  void unlock() { locked_.store(false, std::memory_order_release); }
130 
131  private:
132  std::atomic<bool> locked_;
133 };

141 //
142 // Inspired by Guava: https://github.com/google/guava/wiki/StripedExplained
143 // A striped Lock. This offers the underlying lock striping similar
144 // to that of ConcurrentHashMap in a reusable form, and extends it for
145 // semaphores and read-write locks. Conceptually, lock striping is the technique
146 // of dividing a lock into many <i>stripes</i>, increasing the granularity of a
147 // single lock and allowing independent operations to lock different stripes and
148 // proceed concurrently, instead of creating contention for a single lock.
149 //
150 template <class T, class P>
151 class Striped {
152  public:
153  Striped(size_t stripes, std::function<uint64_t(const P &)> hash)
154  : stripes_(stripes), hash_(hash) {
155 
156  locks_ = reinterpret_cast<LockData<T> *>(
157  port::cacheline_aligned_alloc(sizeof(LockData<T>) * stripes));
158  for (size_t i = 0; i < stripes; i++) {
159  new (&locks_[i]) LockData<T>();
160  }
161 
162  }
163 
164  virtual ~Striped() {
165  if (locks_ != nullptr) {
166  assert(stripes_ > 0);
167  for (size_t i = 0; i < stripes_; i++) {
168  locks_[i].~LockData<T>();
169  }
170  port::cacheline_aligned_free(locks_);
171  }
172  }
173 
174  T *get(const P &key) {
175  uint64_t h = hash_(key);
176  size_t index = h % stripes_;
177  return &reinterpret_cast<LockData<T> *>(&locks_[index])->lock_;
178  }
179 
180  private:
181  size_t stripes_;
182  LockData<T> *locks_;
183  std::function<uint64_t(const P &)> hash_;
184 };
