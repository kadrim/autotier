57 // The MultiGetContext class is a container for the sorted list of keys that
58 // we need to lookup in a batch. Its main purpose is to make batch execution
59 // easier by allowing various stages of the MultiGet lookups to operate on
60 // subsets of keys, potentially non-contiguous. In order to accomplish this,
61 // it defines the following classes -
62 //
63 // MultiGetContext::Range
64 // MultiGetContext::Range::Iterator
65 // MultiGetContext::Range::IteratorWrapper
66 //
67 // Here is an example of how this can be used -
68 //
69 // {
70 // MultiGetContext ctx(...);
71 // MultiGetContext::Range range = ctx.GetMultiGetRange();
72 //
73 // // Iterate to determine some subset of the keys
74 // MultiGetContext::Range::Iterator start = range.begin();
75 // MultiGetContext::Range::Iterator end = ...;
76 //
77 // // Make a new range with a subset of keys
78 // MultiGetContext::Range subrange(range, start, end);
79 //
80 // // Define an auxillary vector, if needed, to hold additional data for
81 // // each key
82 // std::array<Foo, MultiGetContext::MAX_BATCH_SIZE> aux;
83 //
84 // // Iterate over the subrange and the auxillary vector simultaneously
85 // MultiGetContext::Range::Iterator iter = subrange.begin();
86 // for (; iter != subrange.end(); ++iter) {
87 // KeyContext& key = *iter;
88 // Foo& aux_key = aux_iter[iter.index()];
89 // ...
90 // }
91 // }
92 class MultiGetContext {
93  public:
94  // Limit the number of keys in a batch to this number. Benchmarks show that
95  // there is negligible benefit for batches exceeding this. Keeping this < 32
96  // simplifies iteration, as well as reduces the amount of stack allocations
97  // that need to be performed
98  static const int MAX_BATCH_SIZE = 32;
99 
100  static_assert(MAX_BATCH_SIZE < 64, "MAX_BATCH_SIZE cannot exceed 63");
101 
102  MultiGetContext(autovector<KeyContext*, MAX_BATCH_SIZE>* sorted_keys,
103  size_t begin, size_t num_keys, SequenceNumber snapshot,
104  const ReadOptions& read_opts)
105  : num_keys_(num_keys),
106  value_mask_(0),
107  value_size_(0),
108  lookup_key_ptr_(reinterpret_cast<LookupKey*>(lookup_key_stack_buf)) {
109  assert(num_keys <= MAX_BATCH_SIZE);
110  if (num_keys > MAX_LOOKUP_KEYS_ON_STACK) {
111  lookup_key_heap_buf.reset(new char[sizeof(LookupKey) * num_keys]);
112  lookup_key_ptr_ = reinterpret_cast<LookupKey*>(
113  lookup_key_heap_buf.get());
114  }
115 
116  for (size_t iter = 0; iter != num_keys_; ++iter) {
117  // autovector may not be contiguous storage, so make a copy
118  sorted_keys_[iter] = (*sorted_keys)[begin + iter];
119  sorted_keys_[iter]->lkey = new (&lookup_key_ptr_[iter])
120  LookupKey(*sorted_keys_[iter]->key, snapshot, read_opts.timestamp);
121  sorted_keys_[iter]->ukey_with_ts = sorted_keys_[iter]->lkey->user_key();
122  sorted_keys_[iter]->ukey_without_ts = StripTimestampFromUserKey(
123  sorted_keys_[iter]->lkey->user_key(),
124  read_opts.timestamp == nullptr ? 0 : read_opts.timestamp->size());
125  sorted_keys_[iter]->ikey = sorted_keys_[iter]->lkey->internal_key();
126  }
127  }
128 
129  ~MultiGetContext() {
130  for (size_t i = 0; i < num_keys_; ++i) {
131  lookup_key_ptr_[i].~LookupKey();
132  }
133  }
134 
135  private:
136  static const int MAX_LOOKUP_KEYS_ON_STACK = 16;
137  alignas(alignof(LookupKey))
138  char lookup_key_stack_buf[sizeof(LookupKey) * MAX_LOOKUP_KEYS_ON_STACK];
139  std::array<KeyContext*, MAX_BATCH_SIZE> sorted_keys_;
140  size_t num_keys_;
141  uint64_t value_mask_;
142  uint64_t value_size_;
143  std::unique_ptr<char[]> lookup_key_heap_buf;
144  LookupKey* lookup_key_ptr_;

147  // MultiGetContext::Range - Specifies a range of keys, by start and end index,
148  // from the parent MultiGetContext. Each range contains a bit vector that
149  // indicates whether the corresponding keys need to be processed or skipped.
150  // A Range object can be copy constructed, and the new object inherits the
151  // original Range's bit vector. This is useful for progressively skipping
152  // keys as the lookup goes through various stages. For example, when looking
153  // up keys in the same SST file, a Range is created excluding keys not
154  // belonging to that file. A new Range is then copy constructed and individual
155  // keys are skipped based on bloom filter lookup.
156  class Range {
157  public:
158  // MultiGetContext::Range::Iterator - A forward iterator that iterates over
159  // non-skippable keys in a Range, as well as keys whose final value has been
160  // found. The latter is tracked by MultiGetContext::value_mask_
161  class Iterator {
162  public:
163  // -- iterator traits
164  using self_type = Iterator;
165  using value_type = KeyContext;
166  using reference = KeyContext&;
167  using pointer = KeyContext*;
168  using difference_type = int;
169  using iterator_category = std::forward_iterator_tag;
170 
171  Iterator(const Range* range, size_t idx)
172  : range_(range), ctx_(range->ctx_), index_(idx) {
173  while (index_ < range_->end_ &&
174  (uint64_t{1} << index_) &
175  (range_->ctx_->value_mask_ | range_->skip_mask_))
176  index_++;
177  }
178 
179  Iterator(const Iterator&) = default;
180  Iterator& operator=(const Iterator&) = default;
181 
182  Iterator& operator++() {
183  while (++index_ < range_->end_ &&
184  (uint64_t{1} << index_) &
185  (range_->ctx_->value_mask_ | range_->skip_mask_))
186  ;
187  return *this;
188  }
189 
190  bool operator==(Iterator other) const {
191  assert(range_->ctx_ == other.range_->ctx_);
192  return index_ == other.index_;
193  }
194 
195  bool operator!=(Iterator other) const {
196  assert(range_->ctx_ == other.range_->ctx_);
197  return index_ != other.index_;
198  }

200  KeyContext& operator*() {
201  assert(index_ < range_->end_ && index_ >= range_->start_);
202  return *(ctx_->sorted_keys_[index_]);
203  }

206  assert(index_ < range_->end_ && index_ >= range_->start_);
207  return ctx_->sorted_keys_[index_];
208  }

280  return (((uint64_t{1} << end_) - 1) & ~((uint64_t{1} << start_) - 1) &
