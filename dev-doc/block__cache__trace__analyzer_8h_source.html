68  void AddAccess(const BlockCacheTraceRecord& access,
69  uint64_t access_sequnce_number) {
70  if (block_size != 0 && access.block_size != 0) {
71  assert(block_size == access.block_size);
72  }
73  if (num_keys != 0 && access.num_keys_in_block != 0) {
74  assert(num_keys == access.num_keys_in_block);
75  }
76  if (first_access_time == 0) {
77  first_access_time = access.access_timestamp;
78  }
79  table_id = BlockCacheTraceHelper::GetTableId(access);
80  block_offset = BlockCacheTraceHelper::GetBlockOffsetInFile(access);
81  last_access_time = access.access_timestamp;
82  block_size = access.block_size;
83  caller_num_access_map[access.caller]++;
84  num_accesses++;
85  // access.access_timestamp is in microsecond.
86  const uint64_t timestamp_in_seconds =
87  access.access_timestamp / kMicrosInSecond;
88  caller_num_accesses_timeline[access.caller][timestamp_in_seconds] += 1;
89  // Populate the feature vectors.
90  access_sequence_number_timeline.push_back(access_sequnce_number);
91  caller_access_sequence__number_timeline[access.caller].push_back(
92  access_sequnce_number);
93  access_timeline.push_back(access.access_timestamp);
94  caller_access_timeline[access.caller].push_back(access.access_timestamp);
95  if (BlockCacheTraceHelper::IsGetOrMultiGetOnDataBlock(access.block_type,
96  access.caller)) {
97  num_keys = access.num_keys_in_block;
98  if (access.referenced_key_exist_in_block == Boolean::kTrue) {
99  if (key_num_access_map.find(access.referenced_key) ==
100  key_num_access_map.end()) {
101  referenced_data_size += access.referenced_data_size;
102  }
103  key_num_access_map[access.referenced_key][access.caller]++;
104  num_referenced_key_exist_in_block++;
105  if (referenced_data_size > block_size && block_size != 0) {
106  ParsedInternalKey internal_key;
107  Status s = ParseInternalKey(access.referenced_key, &internal_key,
108  false /* log_err_key */); // TODO
109  assert(s.ok()); // TODO
110  }
111  } else {
112  non_exist_key_num_access_map[access.referenced_key][access.caller]++;
113  }
114  }
115  }
116 };
