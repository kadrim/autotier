18 // This is the class to generate the file checksum based on Crc32. It
19 // will be used as the default checksum method for SST file checksum
20 class FileChecksumGenCrc32c : public FileChecksumGenerator {
21  public:
22  FileChecksumGenCrc32c(const FileChecksumGenContext& /*context*/) {
23  checksum_ = 0;
24  }
25 
26  void Update(const char* data, size_t n) override {
27  checksum_ = crc32c::Extend(checksum_, data, n);
28  }
29 
30  void Finalize() override {
31  assert(checksum_str_.empty());
32  // Store as big endian raw bytes
33  PutFixed32(&checksum_str_, EndianSwapValue(checksum_));
34  }
35 
36  std::string GetChecksum() const override {
37  assert(!checksum_str_.empty());
38  return checksum_str_;
39  }
40 
41  const char* Name() const override { return "FileChecksumCrc32c"; }
42 
43  private:
44  uint32_t checksum_;
45  std::string checksum_str_;
46 };
