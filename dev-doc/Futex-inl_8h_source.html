17 template <typename TargetClock, typename Clock, typename Duration>
18 typename TargetClock::time_point time_point_conv(
19  std::chrono::time_point<Clock, Duration> const& time) {
20  using std::chrono::duration_cast;
21  using TimePoint = std::chrono::time_point<Clock, Duration>;
22  using TargetDuration = typename TargetClock::duration;
23  using TargetTimePoint = typename TargetClock::time_point;
24  if (time == TimePoint::max()) {
25  return TargetTimePoint::max();
26  } else if (std::is_same<Clock, TargetClock>::value) {
27  // in place of time_point_cast, which cannot compile without if-constexpr
28  auto const delta = time.time_since_epoch();
29  return TargetTimePoint(duration_cast<TargetDuration>(delta));
30  } else {
31  // different clocks with different epochs, so non-optimal case
32  auto const delta = time - Clock::now();
33  return TargetClock::now() + duration_cast<TargetDuration>(delta);
34  }
35 }

100 template <typename Futex, class Clock, class Duration>
101 FutexResult futexWaitUntil(
102  const Futex* futex,
103  uint32_t expected,
104  std::chrono::time_point<Clock, Duration> const& deadline,
105  uint32_t waitMask) {
106  using Target = typename std::conditional<
107  Clock::is_steady,
108  std::chrono::steady_clock,
109  std::chrono::system_clock>::type;
110  auto const converted = time_point_conv<Target>(deadline);
111  return converted == Target::time_point::max()
112  ? futexWaitImpl(futex, expected, nullptr, nullptr, waitMask)
113  : futexWaitImpl(futex, expected, converted, waitMask);
114 }
