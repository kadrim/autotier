20 // Fast implementation of floor(log2(v)). Undefined for 0 or negative
21 // numbers (in case of signed type).
22 template <typename T>
23 inline int FloorLog2(T v) {
24  static_assert(std::is_integral<T>::value, "non-integral type");
25  assert(v > 0);
26 #ifdef _MSC_VER
27  static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
28  unsigned long idx = 0;
29  if (sizeof(T) <= sizeof(uint32_t)) {
30  _BitScanReverse(&idx, static_cast<uint32_t>(v));
31  } else {
32 #if defined(_M_X64) || defined(_M_ARM64)
33  _BitScanReverse64(&idx, static_cast<uint64_t>(v));
34 #else
35  const auto vh = static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32);
36  if (vh != 0) {
37  _BitScanReverse(&idx, static_cast<uint32_t>(vh));
38  idx += 32;
39  } else {
40  _BitScanReverse(&idx, static_cast<uint32_t>(v));
41  }
42 #endif
43  }
44  return idx;
45 #else
46  static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
47  if (sizeof(T) <= sizeof(unsigned int)) {
48  int lz = __builtin_clz(static_cast<unsigned int>(v));
49  return int{sizeof(unsigned int)} * 8 - 1 - lz;
50  } else if (sizeof(T) <= sizeof(unsigned long)) {
51  int lz = __builtin_clzl(static_cast<unsigned long>(v));
52  return int{sizeof(unsigned long)} * 8 - 1 - lz;
53  } else {
54  int lz = __builtin_clzll(static_cast<unsigned long long>(v));
55  return int{sizeof(unsigned long long)} * 8 - 1 - lz;
56  }
57 #endif
58 }

60 // Number of low-order zero bits before the first 1 bit. Undefined for 0.
61 template <typename T>
62 inline int CountTrailingZeroBits(T v) {
63  static_assert(std::is_integral<T>::value, "non-integral type");
64  assert(v != 0);
65 #ifdef _MSC_VER
66  static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
67  unsigned long tz = 0;
68  if (sizeof(T) <= sizeof(uint32_t)) {
69  _BitScanForward(&tz, static_cast<uint32_t>(v));
70  } else {
71 #if defined(_M_X64) || defined(_M_ARM64)
72  _BitScanForward64(&tz, static_cast<uint64_t>(v));
73 #else
74  _BitScanForward(&tz, static_cast<uint32_t>(v));
75  if (tz == 0) {
76  _BitScanForward(&tz,
77  static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32));
78  tz += 32;
79  }
80 #endif
81  }
82  return static_cast<int>(tz);
83 #else
84  static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
85  if (sizeof(T) <= sizeof(unsigned int)) {
86  return __builtin_ctz(static_cast<unsigned int>(v));
87  } else if (sizeof(T) <= sizeof(unsigned long)) {
88  return __builtin_ctzl(static_cast<unsigned long>(v));
89  } else {
90  return __builtin_ctzll(static_cast<unsigned long long>(v));
91  }
92 #endif
93 }

96 namespace detail {
97 template <typename T>
98 int BitsSetToOneFallback(T v) {
99  const int kBits = static_cast<int>(sizeof(T)) * 8;
100  static_assert((kBits & (kBits - 1)) == 0, "must be power of two bits");
101  // we static_cast these bit patterns in order to truncate them to the correct
102  // size
103  v = static_cast<T>(v - ((v >> 1) & static_cast<T>(0x5555555555555555ull)));
104  v = static_cast<T>((v & static_cast<T>(0x3333333333333333ull)) +
105  ((v >> 2) & static_cast<T>(0x3333333333333333ull)));
106  v = static_cast<T>((v + (v >> 4)) & static_cast<T>(0x0F0F0F0F0F0F0F0Full));
107  for (int shift_bits = 8; shift_bits < kBits; shift_bits <<= 1) {
108  v += static_cast<T>(v >> shift_bits);
109  }
110  // we want the bottom "slot" that's big enough to represent a value up to
111  // (and including) kBits.
112  return static_cast<int>(v & static_cast<T>(kBits | (kBits - 1)));
113 }

118 // Number of bits set to 1. Also known as "population count".
119 template <typename T>
120 inline int BitsSetToOne(T v) {
121  static_assert(std::is_integral<T>::value, "non-integral type");
122 #ifdef _MSC_VER
123  static_assert(sizeof(T) <= sizeof(uint64_t), "type too big");
124  if (sizeof(T) < sizeof(uint32_t)) {
125  // This bit mask is to avoid a compiler warning on unused path
126  constexpr auto mm = 8 * sizeof(uint32_t) - 1;
127  // The bit mask is to neutralize sign extension on small signed types
128  constexpr uint32_t m = (uint32_t{1} << ((8 * sizeof(T)) & mm)) - 1;
129 #if defined(_M_X64) || defined(_M_IX86)
130  return static_cast<int>(__popcnt(static_cast<uint32_t>(v) & m));
131 #else
132  return static_cast<int>(detail::BitsSetToOneFallback(v) & m);
133 #endif
134  } else if (sizeof(T) == sizeof(uint32_t)) {
135 #if defined(_M_X64) || defined(_M_IX86)
136  return static_cast<int>(__popcnt(static_cast<uint32_t>(v)));
137 #else
138  return detail::BitsSetToOneFallback(static_cast<uint32_t>(v));
139 #endif
140  } else {
141 #ifdef _M_X64
142  return static_cast<int>(__popcnt64(static_cast<uint64_t>(v)));
143 #elif defined(_M_IX86)
144  return static_cast<int>(
145  __popcnt(static_cast<uint32_t>(static_cast<uint64_t>(v) >> 32) +
146  __popcnt(static_cast<uint32_t>(v))));
147 #else
148  return detail::BitsSetToOneFallback(static_cast<uint64_t>(v));
149 #endif
150  }
151 #else
152  static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
153  if (sizeof(T) < sizeof(unsigned int)) {
154  // This bit mask is to avoid a compiler warning on unused path
155  constexpr auto mm = 8 * sizeof(unsigned int) - 1;
156  // This bit mask is to neutralize sign extension on small signed types
157  constexpr unsigned int m = (1U << ((8 * sizeof(T)) & mm)) - 1;
158  return __builtin_popcount(static_cast<unsigned int>(v) & m);
159  } else if (sizeof(T) == sizeof(unsigned int)) {
160  return __builtin_popcount(static_cast<unsigned int>(v));
161  } else if (sizeof(T) <= sizeof(unsigned long)) {
162  return __builtin_popcountl(static_cast<unsigned long>(v));
163  } else {
164  return __builtin_popcountll(static_cast<unsigned long long>(v));
165  }
166 #endif
167 }

169 template <typename T>
170 inline int BitParity(T v) {
171  static_assert(std::is_integral<T>::value, "non-integral type");
172 #ifdef _MSC_VER
173  // bit parity == oddness of popcount
174  return BitsSetToOne(v) & 1;
175 #else
176  static_assert(sizeof(T) <= sizeof(unsigned long long), "type too big");
177  if (sizeof(T) <= sizeof(unsigned int)) {
178  // On any sane systen, potential sign extension here won't change parity
179  return __builtin_parity(static_cast<unsigned int>(v));
180  } else if (sizeof(T) <= sizeof(unsigned long)) {
181  return __builtin_parityl(static_cast<unsigned long>(v));
182  } else {
183  return __builtin_parityll(static_cast<unsigned long long>(v));
184  }
185 #endif
186 }
