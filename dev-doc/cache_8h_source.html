10 // A Cache is an interface that maps keys to values. It has internal
11 // synchronization and may be safely accessed concurrently from
12 // multiple threads. It may automatically evict entries to make room
13 // for new entries. Values have a specified charge against the cache
14 // capacity. For example, a cache where the values are variable
15 // length strings, may use the length of the string as the charge for
16 // the string.
17 //
18 // A builtin cache implementation with a least-recently-used eviction
19 // policy is provided. Clients may use their own implementations if
20 // they want something more sophisticated (like scan-resistance, a
21 // custom eviction policy, variable cache sizing, etc.)

146  // Depending on implementation, cache entries with high priority could be less

150  // A set of callbacks to allow objects in the primary block cache to be
151  // be persisted in a secondary cache. The purpose of the secondary cache
152  // is to support other ways of caching the object, such as persistent or
153  // compressed data, that may require the object to be parsed and transformed
154  // in some way. Since the primary cache holds C++ objects and the secondary
155  // cache may only hold flat data that doesn't need relocation, these
156  // callbacks need to be provided by the user of the block
157  // cache to do the conversion.
158  // The CacheItemHelper is passed to Insert() and Lookup(). It has pointers
159  // to callback functions for size, saving and deletion of the
160  // object. The callbacks are defined in C-style in order to make them
161  // stateless and not add to the cache metadata size.
162  // Saving multiple std::function objects will take up 32 bytes per
163  // function, even if its not bound to an object and does no capture.
164  //
165  // All the callbacks are C-style function pointers in order to simplify
166  // lifecycle management. Objects in the cache can outlive the parent DB,
167  // so anything required for these operations should be contained in the
168  // object itself.
169  //
170  // The SizeCallback takes a void* pointer to the object and returns the size
171  // of the persistable data. It can be used by the secondary cache to allocate
172  // memory if needed.
173  using SizeCallback = size_t (*)(void* obj);
174 
175  // The SaveToCallback takes a void* object pointer and saves the persistable
176  // data into a buffer. The secondary cache may decide to not store it in a
177  // contiguous buffer, in which case this callback will be called multiple
178  // times with increasing offset
179  using SaveToCallback = Status (*)(void* from_obj, size_t from_offset,
180  size_t length, void* out);
181 
182  // A function pointer type for custom destruction of an entry's
183  // value. The Cache is responsible for copying and reclaiming space
184  // for the key, but values are managed by the caller.
185  using DeleterFn = void (*)(const Slice& key, void* value);
186 
187  // A struct with pointers to helper functions for spilling items from the
188  // cache into the secondary cache. May be extended in the future. An
189  // instance of this struct is expected to outlive the cache.
190  struct CacheItemHelper {
191  SizeCallback size_cb;
192  SaveToCallback saveto_cb;
193  DeleterFn del_cb;
194 
195  CacheItemHelper() : size_cb(nullptr), saveto_cb(nullptr), del_cb(nullptr) {}
196  CacheItemHelper(SizeCallback _size_cb, SaveToCallback _saveto_cb,
197  DeleterFn _del_cb)
198  : size_cb(_size_cb), saveto_cb(_saveto_cb), del_cb(_del_cb) {}
199  };

387  // EXPERIMENTAL
388  // The following APIs are experimental and might change in the future.
389  // The Insert and Lookup APIs below are intended to allow cached objects
390  // to be demoted/promoted between the primary block cache and a secondary
391  // cache. The secondary cache could be a non-volatile cache, and will
392  // likely store the object in a different representation more suitable
393  // for on disk storage. They rely on a per object CacheItemHelper to do
394  // the conversions.
395  // The secondary cache may persist across process and system restarts,
396  // and may even be moved between hosts. Therefore, the cache key must
397  // be repeatable across restarts/reboots, and globally unique if
398  // multiple DBs share the same cache and the set of DBs can change
399  // over time.
400 
401  // Insert a mapping from key->value into the cache and assign it
402  // the specified charge against the total cache capacity.
403  // If strict_capacity_limit is true and cache reaches its full capacity,
404  // return Status::Incomplete.
405  //
406  // The helper argument is saved by the cache and will be used when the
407  // inserted object is evicted or promoted to the secondary cache. It,
408  // therefore, must outlive the cache.
409  //
410  // If handle is not nullptr, returns a handle that corresponds to the
411  // mapping. The caller must call this->Release(handle) when the returned
412  // mapping is no longer needed. In case of error caller is responsible to
413  // cleanup the value (i.e. calling "deleter").
414  //
415  // If handle is nullptr, it is as if Release is called immediately after
416  // insert. In case of error value will be cleanup.
417  //
418  // Regardless of whether the item was inserted into the cache,
419  // it will attempt to insert it into the secondary cache if one is
420  // configured, and the helper supports it.
421  // The cache implementation must support a secondary cache, otherwise
422  // the item is only inserted into the primary cache. It may
423  // defer the insertion to the secondary cache as it sees fit.
424  //
425  // When the inserted entry is no longer needed, the key and
426  // value will be passed to "deleter".
427  virtual Status Insert(const Slice& key, void* value,
428  const CacheItemHelper* helper, size_t charge,
429  Handle** handle = nullptr,
430  Priority priority = Priority::LOW) {
431  if (!helper) {
432  return Status::InvalidArgument();
433  }
434  return Insert(key, value, charge, helper->del_cb, handle, priority);
435  }

437  // Lookup the key in the primary and secondary caches (if one is configured).
438  // The create_cb callback function object will be used to contruct the
439  // cached object.
440  // If none of the caches have the mapping for the key, returns nullptr.
441  // Else, returns a handle that corresponds to the mapping.
442  //
443  // This call may promote the object from the secondary cache (if one is
444  // configured, and has the given key) to the primary cache.
445  //
446  // The helper argument should be provided if the caller wants the lookup
447  // to include the secondary cache (if one is configured) and the object,
448  // if it exists, to be promoted to the primary cache. The helper may be
449  // saved and used later when the object is evicted. Therefore, it must
450  // outlive the cache.
451  //
452  // The handle returned may not be ready. The caller should call IsReady()
453  // to check if the item value is ready, and call Wait() or WaitAll() if
454  // its not ready. The caller should then call Value() to check if the
455  // item was successfully retrieved. If unsuccessful (perhaps due to an
456  // IO error), Value() will return nullptr.
457  virtual Handle* Lookup(const Slice& key, const CacheItemHelper* /*helper_cb*/,
458  const CreateCallback& /*create_cb*/,
459  Priority /*priority*/, bool /*wait*/,
460  Statistics* stats = nullptr) {
461  return Lookup(key, stats);
462  }
