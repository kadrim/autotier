29  // Verifies there have been no commits to this key in the db since this
30  // sequence number.
31  //
32  // If cache_only is true, then this function will not attempt to read any
33  // SST files. This will make it more likely this function will
34  // return an error if it is unable to determine if there are any conflicts.
35  //
36  // See comment of CheckKey() for explanation of `snap_seq`, `snap_checker`
37  // and `min_uncommitted`.
38  //
39  // Returns OK on success, BUSY if there is a conflicting write, or other error
40  // status for any unexpected errors.
41  static Status CheckKeyForConflicts(
42  DBImpl* db_impl, ColumnFamilyHandle* column_family,
43  const std::string& key, SequenceNumber snap_seq, bool cache_only,
44  ReadCallback* snap_checker = nullptr,
45  SequenceNumber min_uncommitted = kMaxSequenceNumber);
46 
47  // For each key,SequenceNumber pair tracked by the LockTracker, this function
48  // will verify there have been no writes to the key in the db since that
49  // sequence number.
50  //
51  // Returns OK on success, BUSY if there is a conflicting write, or other error
52  // status for any unexpected errors.
53  //
54  // REQUIRED:
55  // This function should only be called on the write thread or if the
56  // mutex is held.
57  // tracker must support point lock.
58  static Status CheckKeysForConflicts(DBImpl* db_impl,
59  const LockTracker& tracker,
60  bool cache_only);
61 
62  private:
63  // If `snap_checker` == nullptr, writes are always commited in sequence number
64  // order. All sequence number <= `snap_seq` will not conflict with any
65  // write, and all keys > `snap_seq` of `key` will trigger conflict.
66  // If `snap_checker` != nullptr, writes may not commit in sequence number
67  // order. In this case `min_uncommitted` is a lower bound.
68  // seq < `min_uncommitted`: no conflict
69  // seq > `snap_seq`: applicable to conflict
70  // `min_uncommitted` <= seq <= `snap_seq`: call `snap_checker` to determine.
71  static Status CheckKey(DBImpl* db_impl, SuperVersion* sv,
72  SequenceNumber earliest_seq, SequenceNumber snap_seq,
73  const std::string& key, bool cache_only,
74  ReadCallback* snap_checker = nullptr,
75  SequenceNumber min_uncommitted = kMaxSequenceNumber);
