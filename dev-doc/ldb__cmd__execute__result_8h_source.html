31  std::string ToString() {
32  std::string ret;
33  switch (state_) {
34  case EXEC_SUCCEED:
35  break;
36  case EXEC_FAILED:

46  }

51  }

53  bool IsSucceed() { return state_ == EXEC_SUCCEED; }
54 
55  bool IsNotStarted() { return state_ == EXEC_NOT_STARTED; }
56
