197 // This class stores references to all the immutable memtables.
198 // The memtables are flushed to L0 as soon as possible and in
199 // any order. If there are more than one immutable memtable, their
200 // flushes can occur concurrently. However, they are 'committed'
201 // to the manifest in FIFO order to maintain correctness and
202 // recoverability from a crash.
203 //
204 //

371  void AssignAtomicFlushSeq(const SequenceNumber& seq) {
372  const auto& memlist = current_->memlist_;
373  // Scan the memtable list from new to old
374  for (auto it = memlist.begin(); it != memlist.end(); ++it) {
375  MemTable* mem = *it;
376  if (mem->atomic_flush_seqno_ == kMaxSequenceNumber) {
377  mem->atomic_flush_seqno_ = seq;
378  } else {
379  // Earlier memtables must have been assigned a atomic flush seq, no
380  // need to continue scan.
381  break;
382  }
383  }
384  }
