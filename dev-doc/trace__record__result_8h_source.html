55 // Base class for the results from the trace record execution handler (created
56 // by TraceRecord::NewExecutionHandler()).
57 //
58 // The actual execution status or returned values may be hidden from
59 // TraceRecord::Handler::Handle and TraceRecord::Accept. For example, a
60 // GetQueryTraceRecord's execution calls DB::Get() internally. DB::Get() may
61 // return Status::NotFound() but TraceRecord::Handler::Handle() or
62 // TraceRecord::Accept() will still return Status::OK(). The actual status from
63 // DB::Get() and the returned value string may be saved in a
64 // SingleValueTraceExecutionResult.
65 class TraceExecutionResult : public TraceRecordResult {
66  public:
67  TraceExecutionResult(uint64_t start_timestamp, uint64_t end_timestamp,
68  TraceType trace_type);
69 
70  // Execution start/end timestamps and request latency in microseconds.
71  virtual uint64_t GetStartTimestamp() const;
72  virtual uint64_t GetEndTimestamp() const;
73  inline uint64_t GetLatency() const {
74  return GetEndTimestamp() - GetStartTimestamp();
75  }
76 
77  private:
78  uint64_t ts_start_;
79  uint64_t ts_end_;
80 };
