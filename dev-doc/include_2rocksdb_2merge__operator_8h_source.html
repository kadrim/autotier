21 // The Merge Operator
22 //
23 // Essentially, a MergeOperator specifies the SEMANTICS of a merge, which only
24 // client knows. It could be numeric addition, list append, string
25 // concatenation, edit data structure, ... , anything.
26 // The library, on the other hand, is concerned with the exercise of this
27 // interface, at the right time (during get, iteration, compaction...)
28 //
29 // To use merge, the client needs to provide an object implementing one of
30 // the following interfaces:
31 // a) AssociativeMergeOperator - for most simple semantics (always take
32 // two values, and merge them into one value, which is then put back
33 // into rocksdb); numeric addition and string concatenation are examples;
34 //
35 // b) MergeOperator - the generic class for all the more abstract / complex
36 // operations; one method (FullMergeV2) to merge a Put/Delete value with a
37 // merge operand; and another method (PartialMerge) that merges multiple
38 // operands together. this is especially useful if your key values have
39 // complex structures but you would still like to support client-specific
40 // incremental updates.
41 //
42 // AssociativeMergeOperator is simpler to implement. MergeOperator is simply
43 // more powerful.
44 //
45 // Refer to rocksdb-merge wiki for more details and example implementations.
46 //
47 class MergeOperator : public Customizable {
48  public:
49  virtual ~MergeOperator() {}
50  static const char* Type() { return "MergeOperator"; }
51  static Status CreateFromString(const ConfigOptions& opts,
52  const std::string& id,
53  std::shared_ptr<MergeOperator>* result);

55  // Gives the client a way to express the read -> modify -> write semantics
56  // key: (IN) The key that's associated with this merge operation.
57  // Client could multiplex the merge operator based on it
58  // if the key space is partitioned and different subspaces
59  // refer to different types of data which have different
60  // merge operation semantics
61  // existing: (IN) null indicates that the key does not exist before this op
62  // operand_list:(IN) the sequence of merge operations to apply, front() first.
63  // new_value:(OUT) Client is responsible for filling the merge result here.
64  // The string that new_value is pointing to will be empty.
65  // logger: (IN) Client could use this to log errors during merge.
66  //
67  // Return true on success.
68  // All values passed in will be client-specific values. So if this method
69  // returns false, it is because client specified bad data or there was
70  // internal corruption. This will be treated as an error by the library.
71  //
72  // Also make use of the *logger for error messages.
73  virtual bool FullMerge(const Slice& /*key*/, const Slice* /*existing_value*/,
74  const std::deque<std::string>& /*operand_list*/,
75  std::string* /*new_value*/, Logger* /*logger*/) const {
76  // deprecated, please use FullMergeV2()
77  assert(false);
78  return false;
79  }

116  // This function applies a stack of merge operands in chronological order
117  // on top of an existing value. There are two ways in which this method is
118  // being used:
119  // a) During Get() operation, it used to calculate the final value of a key
120  // b) During compaction, in order to collapse some operands with the based
121  // value.
122  //
123  // Note: The name of the method is somewhat misleading, as both in the cases
124  // of Get() or compaction it may be called on a subset of operands:
125  // K: 0 +1 +2 +7 +4 +5 2 +1 +2
126  // ^
127  // |
128  // snapshot
129  // In the example above, Get(K) operation will call FullMerge with a base
130  // value of 2 and operands [+1, +2]. Compaction process might decide to
131  // collapse the beginning of the history up to the snapshot by performing
132  // full Merge with base value of 0 and operands [+1, +2, +7, +4].
133  virtual bool FullMergeV2(const MergeOperationInput& merge_in,
134  MergeOperationOutput* merge_out) const;
135 
136  // This function performs merge(left_op, right_op)
137  // when both the operands are themselves merge operation types
138  // that you would have passed to a DB::Merge() call in the same order
139  // (i.e.: DB::Merge(key,left_op), followed by DB::Merge(key,right_op)).
140  //
141  // PartialMerge should combine them into a single merge operation that is
142  // saved into *new_value, and then it should return true.
143  // *new_value should be constructed such that a call to
144  // DB::Merge(key, *new_value) would yield the same result as a call
145  // to DB::Merge(key, left_op) followed by DB::Merge(key, right_op).
146  //
147  // The string that new_value is pointing to will be empty.
148  //
149  // The default implementation of PartialMergeMulti will use this function
150  // as a helper, for backward compatibility. Any successor class of
151  // MergeOperator should either implement PartialMerge or PartialMergeMulti,
152  // although implementing PartialMergeMulti is suggested as it is in general
153  // more effective to merge multiple operands at a time instead of two
154  // operands at a time.
155  //
156  // If it is impossible or infeasible to combine the two operations,
157  // leave new_value unchanged and return false. The library will
158  // internally keep track of the operations, and apply them in the
159  // correct order once a base-value (a Put/Delete/End-of-Database) is seen.
160  //
161  // TODO: Presently there is no way to differentiate between error/corruption
162  // and simply "return false". For now, the client should simply return
163  // false in any case it cannot perform partial-merge, regardless of reason.
164  // If there is corruption in the data, handle it in the FullMergeV2() function
165  // and return false there. The default implementation of PartialMerge will
166  // always return false.
167  virtual bool PartialMerge(const Slice& /*key*/, const Slice& /*left_operand*/,
168  const Slice& /*right_operand*/,
169  std::string* /*new_value*/,
170  Logger* /*logger*/) const {
171  return false;
172  }
173 
174  // This function performs merge when all the operands are themselves merge
175  // operation types that you would have passed to a DB::Merge() call in the
176  // same order (front() first)
177  // (i.e. DB::Merge(key, operand_list[0]), followed by
178  // DB::Merge(key, operand_list[1]), ...)
179  //
180  // PartialMergeMulti should combine them into a single merge operation that is
181  // saved into *new_value, and then it should return true. *new_value should
182  // be constructed such that a call to DB::Merge(key, *new_value) would yield
183  // the same result as sequential individual calls to DB::Merge(key, operand)
184  // for each operand in operand_list from front() to back().
185  //
186  // The string that new_value is pointing to will be empty.
187  //
188  // The PartialMergeMulti function will be called when there are at least two
189  // operands.
190  //
191  // In the default implementation, PartialMergeMulti will invoke PartialMerge
192  // multiple times, where each time it only merges two operands. Developers
193  // should either implement PartialMergeMulti, or implement PartialMerge which
194  // is served as the helper function of the default PartialMergeMulti.
195  virtual bool PartialMergeMulti(const Slice& key,
196  const std::deque<Slice>& operand_list,
197  std::string* new_value, Logger* logger) const;
198 
199  // The name of the MergeOperator. Used to check for MergeOperator
200  // mismatches (i.e., a DB created with one MergeOperator is
201  // accessed using a different MergeOperator)
202  // TODO: the name is currently not stored persistently and thus
203  // no checking is enforced. Client is responsible for providing
204  // consistent MergeOperator between DB opens.
205  virtual const char* Name() const = 0;
206 
207  // Determines whether the PartialMerge can be called with just a single
208  // merge operand.
209  // Override and return true for allowing a single operand. PartialMerge
210  // and PartialMergeMulti should be overridden and implemented
211  // correctly to properly handle a single operand.
212  virtual bool AllowSingleOperand() const { return false; }

214  // Allows to control when to invoke a full merge during Get.
215  // This could be used to limit the number of merge operands that are looked at
216  // during a point lookup, thereby helping in limiting the number of levels to
217  // read from.
218  // Doesn't help with iterators.
219  //
220  // Note: the merge operands are passed to this function in the reversed order
221  // relative to how they were merged (passed to FullMerge or FullMergeV2)
222  // for performance reasons, see also:
223  // https://github.com/facebook/rocksdb/issues/3865
224  virtual bool ShouldMerge(const std::vector<Slice>& /*operands*/) const {
225  return false;
226  }
