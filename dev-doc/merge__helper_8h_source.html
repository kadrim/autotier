57  // Merge entries until we hit
58  // - a corrupted key
59  // - a Put/Delete,
60  // - a different user key,
61  // - a specific sequence number (snapshot boundary),
62  // - REMOVE_AND_SKIP_UNTIL returned from compaction filter,
63  // or - the end of iteration
64  // iter: (IN) points to the first merge type entry
65  // (OUT) points to the first entry not included in the merge process
66  // range_del_agg: (IN) filters merge operands covered by range tombstones.
67  // stop_before: (IN) a sequence number that merge should not cross.
68  // 0 means no restriction
69  // at_bottom: (IN) true if the iterator covers the bottem level, which means
70  // we could reach the start of the history of this user key.
71  // allow_data_in_errors: (IN) if true, data details will be displayed in
72  // error/log messages.
73  //
74  // Returns one of the following statuses:
75  // - OK: Entries were successfully merged.
76  // - MergeInProgress: Put/Delete not encountered, and didn't reach the start
77  // of key's history. Output consists of merge operands only.
78  // - Corruption: Merge operator reported unsuccessful merge or a corrupted
79  // key has been encountered and not expected (applies only when compiling
80  // with asserts removed).
81  // - ShutdownInProgress: interrupted by shutdown (*shutting_down == true).
82  //
83  // REQUIRED: The first key in the input is not corrupted.

98  // Query the merge result
99  // These are valid until the next MergeUntil call
100  // If the merging was successful:
101  // - keys() contains a single element with the latest sequence number of
102  // the merges. The type will be Put or Merge. See IMPORTANT 1 note, below.
103  // - values() contains a single element with the result of merging all the
104  // operands together
105  //
106  // IMPORTANT 1: the key type could change after the MergeUntil call.
107  // Put/Delete + Merge + ... + Merge => Put
108  // Merge + ... + Merge => Merge
109  //
110  // If the merge operator is not associative, and if a Put/Delete is not found
111  // then the merging will be unsuccessful. In this case:
112  // - keys() contains the list of internal keys seen in order of iteration.
113  // - values() contains the list of values (merges) seen in the same order.
114  // values() is parallel to keys() so that the first entry in
115  // keys() is the key associated with the first entry in values()
116  // and so on. These lists will be the same length.
117  // All of these pairs will be merges over the same user key.
118  // See IMPORTANT 2 note below.
119  //
120  // IMPORTANT 2: The entries were traversed in order from BACK to FRONT.
121  // So keys().back() was the first key seen by iterator.
122  // TODO: Re-style this comment to be like the first one
123  const std::deque<std::string>& keys() const { return keys_; }
124  const std::vector<Slice>& values() const {
125  return merge_context_.GetOperands();
126  }
127  uint64_t TotalFilterTime() const { return total_filter_time_; }
128  bool HasOperator() const { return user_merge_operator_ != nullptr; }

130  // If compaction filter returned REMOVE_AND_SKIP_UNTIL, this method will
131  // return true and fill *until with the key to which we should skip.
132  // If true, keys() and values() are empty.
133  bool FilteredUntil(Slice* skip_until) const {
134  if (!has_compaction_filter_skip_until_) {
135  return false;
136  }
137  assert(compaction_filter_ != nullptr);
138  assert(skip_until != nullptr);
139  assert(compaction_filter_skip_until_.Valid());
140  *skip_until = compaction_filter_skip_until_.Encode();
141  return true;
142  }
