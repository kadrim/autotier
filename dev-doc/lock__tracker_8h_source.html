93  // Merges lock requests tracked in the specified tracker into the current
94  // tracker.
95  //
96  // E.g. for point lock, if a key in tracker is not yet tracked,
97  // track this new key; otherwise, merge the tracked information of the key
98  // such as lock's exclusiveness, read/write statistics.
99  //
100  // If this method is not supported, leave it as a no-op.
101  //
102  // REQUIRED: the specified tracker must be of the same concrete class type as
103  // the current tracker.
104  virtual void Merge(const LockTracker& /*tracker*/) = 0;
105 
106  // This is a reverse operation of Merge.
107  //
108  // E.g. for point lock, if a key exists in both current and the sepcified
109  // tracker, then subtract the information (such as read/write statistics) of
110  // the key in the specified tracker from the current tracker.
111  //
112  // If this method is not supported, leave it as a no-op.
113  //
114  // REQUIRED:
115  // The specified tracker must be of the same concrete class type as
116  // the current tracker.
117  // The tracked locks in the specified tracker must be a subset of those
118  // tracked by the current tracker.
119  virtual void Subtract(const LockTracker& /*tracker*/) = 0;
120 
121  // Clears all tracked locks.
122  virtual void Clear() = 0;

124  // Gets the new locks (excluding the locks that have been tracked before the
125  // save point) tracked since the specified save point, the result is stored
126  // in an internally constructed LockTracker and returned.
127  //
128  // save_point_tracker is the tracker used by a SavePoint to track locks
