104  // Allows OS to incrementally sync files to disk while they are being
105  // written, in the background. Issue one request for every bytes_per_sync
106  // written. 0 turns it off.
107  // Default: 0
108  uint64_t bytes_per_sync = 0;
109 
110  // When true, guarantees the file has at most `bytes_per_sync` bytes submitted
111  // for writeback at any given time.
112  //
113  // - If `sync_file_range` is supported it achieves this by waiting for any
114  // prior `sync_file_range`s to finish before proceeding. In this way,
115  // processing (compression, etc.) can proceed uninhibited in the gap
116  // between `sync_file_range`s, and we block only when I/O falls behind.
117  // - Otherwise the `WritableFile::Sync` method is used. Note this mechanism
118  // always blocks, thus preventing the interleaving of I/O and processing.
119  //
120  // Note: Enabling this option does not provide any additional persistence
121  // guarantees, as it may use `sync_file_range`, which does not write out
122  // metadata.
123  //
124  // Default: false
125  bool strict_bytes_per_sync = false;
126 
127  // If true, we will preallocate the file with FALLOC_FL_KEEP_SIZE flag, which
128  // means that file size won't change as part of preallocation.

130  // improve the performance in workloads where you sync the data on every
131  // write. By default, we set it to true for MANIFEST writes and false for

472  using FWType = FunctorWrapper<Args...>;
473  StartThread(
474  [](void* arg) {
475  auto* functor = static_cast<FWType*>(arg);
476  functor->invoke();
477  delete functor;
478  },
479  new FWType(std::function<void(Args...)>(function),
480  std::forward<Args>(args)...));
