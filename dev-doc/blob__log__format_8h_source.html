25 // Format of blob log file header (30 bytes):
26 //
27 // +--------------+---------+---------+-------+-------------+-------------------+
28 // | magic number | version | cf id | flags | compression | expiration range |
29 // +--------------+---------+---------+-------+-------------+-------------------+
30 // | Fixed32 | Fixed32 | Fixed32 | char | char | Fixed64 Fixed64 |
31 // +--------------+---------+---------+-------+-------------+-------------------+
32 //
33 // List of flags:
34 // has_ttl: Whether the file contain TTL data.
35 //
36 // Expiration range in the header is a rough range based on
37 // blob_db_options.ttl_range_secs.
38 struct BlobLogHeader {
39  static constexpr size_t kSize = 30;
40 
41  BlobLogHeader() = default;
42  BlobLogHeader(uint32_t _column_family_id, CompressionType _compression,
43  bool _has_ttl, const ExpirationRange& _expiration_range)
44  : column_family_id(_column_family_id),
45  compression(_compression),
46  has_ttl(_has_ttl),
47  expiration_range(_expiration_range) {}
48 
49  uint32_t version = kVersion1;
50  uint32_t column_family_id = 0;
51  CompressionType compression = kNoCompression;
52  bool has_ttl = false;
53  ExpirationRange expiration_range;
54 
55  void EncodeTo(std::string* dst);
56 
57  Status DecodeFrom(Slice slice);
58 };

60 // Format of blob log file footer (32 bytes):
61 //
62 // +--------------+------------+-------------------+------------+
63 // | magic number | blob count | expiration range | footer CRC |
64 // +--------------+------------+-------------------+------------+
65 // | Fixed32 | Fixed64 | Fixed64 + Fixed64 | Fixed32 |
66 // +--------------+------------+-------------------+------------+
67 //
68 // The footer will be presented only when the blob file is properly closed.
69 //
70 // Unlike the same field in file header, expiration range in the footer is the
71 // range of smallest and largest expiration of the data in this file.
72 struct BlobLogFooter {
73  static constexpr size_t kSize = 32;
74 
75  uint64_t blob_count = 0;
76  ExpirationRange expiration_range = std::make_pair(0, 0);
77  uint32_t crc = 0;
78 
79  void EncodeTo(std::string* dst);
80 
81  Status DecodeFrom(Slice slice);
82 };

84 // Blob record format (32 bytes header + key + value):
85 //
86 // +------------+--------------+------------+------------+----------+---------+-----------+
87 // | key length | value length | expiration | header CRC | blob CRC | key | value |
88 // +------------+--------------+------------+------------+----------+---------+-----------+
89 // | Fixed64 | Fixed64 | Fixed64 | Fixed32 | Fixed32 | key len | value len |
90 // +------------+--------------+------------+------------+----------+---------+-----------+
91 //
92 // If file has has_ttl = false, expiration field is always 0, and the blob
93 // doesn't has expiration.
94 //
95 // Also note that if compression is used, value is compressed value and value
96 // length is compressed value length.
97 //
98 // Header CRC is the checksum of (key_len + val_len + expiration), while
99 // blob CRC is the checksum of (key + value).
100 //
101 // We could use variable length encoding (Varint64) to save more space, but it
102 // make reader more complicated.
103 struct BlobLogRecord {
104  // header include fields up to blob CRC
105  static constexpr size_t kHeaderSize = 32;
106 
107  // Note that the offset field of BlobIndex actually points to the blob value
108  // as opposed to the start of the blob record. The following method can
109  // be used to calculate the adjustment needed to read the blob record header.
110  static constexpr uint64_t CalculateAdjustmentForRecordHeader(
111  uint64_t key_size) {
112  return key_size + kHeaderSize;
113  }
114 
115  uint64_t key_size = 0;
116  uint64_t value_size = 0;
117  uint64_t expiration = 0;
118  uint32_t header_crc = 0;
119  uint32_t blob_crc = 0;
120  Slice key;
121  Slice value;
122  std::unique_ptr<char[]> key_buf;
123  std::unique_ptr<char[]> value_buf;
124 
125  uint64_t record_size() const { return kHeaderSize + key_size + value_size; }
126 
127  void EncodeHeaderTo(std::string* dst);
128 
129  Status DecodeHeaderFrom(Slice src);
130 
131  Status CheckBlobCRC() const;
132 };

134 // Checks whether a blob offset is potentially valid or not.
135 inline bool IsValidBlobOffset(uint64_t value_offset, uint64_t key_size,
136  uint64_t value_size, uint64_t file_size) {
137  if (value_offset <
138  BlobLogHeader::kSize + BlobLogRecord::kHeaderSize + key_size) {
139  return false;
140  }
141 
142  if (value_offset + value_size + BlobLogFooter::kSize > file_size) {
143  return false;
144  }
145 
146  return true;
147 }
