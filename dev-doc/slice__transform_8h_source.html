28 /*
29  * A SliceTransform is a generic pluggable way of transforming one string
30  * to another. Its primary use-case is in configuring rocksdb
31  * to store prefix blooms by setting prefix_extractor in
32  * ColumnFamilyOptions.
33  */
34 class SliceTransform : public Customizable {
35  public:
36  virtual ~SliceTransform(){};

51  // Extract a prefix from a specified key. This method is called when
52  // a key is inserted into the db, and the returned slice is used to
53  // create a bloom filter.
54  virtual Slice Transform(const Slice& key) const = 0;
55 
56  // Determine whether the specified key is compatible with the logic
57  // specified in the Transform method. This method is invoked for every
58  // key that is inserted into the db. If this method returns true,
59  // then Transform is called to translate the key to its prefix and
60  // that returned prefix is inserted into the bloom filter. If this
61  // method returns false, then the call to Transform is skipped and
62  // no prefix is inserted into the bloom filters.
63  //
64  // For example, if the Transform method operates on a fixed length
65  // prefix of size 4, then an invocation to InDomain("abc") returns
66  // false because the specified key length(3) is shorter than the
67  // prefix size of 4.
68  //
69  // Wiki documentation here:
70  // https://github.com/facebook/rocksdb/wiki/Prefix-Seek
71  //
72  virtual bool InDomain(const Slice& key) const = 0;
73 
74  // This is currently not used and remains here for backward compatibility.
75  virtual bool InRange(const Slice& /*dst*/) const { return false; }
76 
77  // Some SliceTransform will have a full length which can be used to
78  // determine if two keys are consecutive. Can be disabled by always
79  // returning 0
80  virtual bool FullLengthEnabled(size_t* /*len*/) const { return false; }
81 
82  // Transform(s)=Transform(`prefix`) for any s with `prefix` as a prefix.
83  //
84  // This function is not used by RocksDB, but for users. If users pass
85  // Options by string to RocksDB, they might not know what prefix extractor
86  // they are using. This function is to help users can determine:
87  // if they want to iterate all keys prefixing `prefix`, whether it is
88  // safe to use prefix bloom filter and seek to key `prefix`.
89  // If this function returns true, this means a user can Seek() to a prefix
90  // using the bloom filter. Otherwise, user needs to skip the bloom filter
91  // by setting ReadOptions.total_order_seek = true.
92  //
93  // Here is an example: Suppose we implement a slice transform that returns
94  // the first part of the string after splitting it using delimiter ",":
95  // 1. SameResultWhenAppended("abc,") should return true. If applying prefix
96  // bloom filter using it, all slices matching "abc:.*" will be extracted
97  // to "abc,", so any SST file or memtable containing any of those key
98  // will not be filtered out.
99  // 2. SameResultWhenAppended("abc") should return false. A user will not
100  // guaranteed to see all the keys matching "abc.*" if a user seek to "abc"
101  // against a DB with the same setting. If one SST file only contains
102  // "abcd,e", the file can be filtered out and the key will be invisible.
103  //
104  // i.e., an implementation always returning false is safe.
105  virtual bool SameResultWhenAppended(const Slice& /*prefix*/) const {
106  return false;
107  }
108 };
