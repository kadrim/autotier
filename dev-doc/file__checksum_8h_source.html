40 // FileChecksumGenerator is the class to generates the checksum value
41 // for each file when the file is written to the file system.
42 // Implementations may assume that
43 // * Finalize is called at most once during the life of the object
44 // * All calls to Update come before Finalize
45 // * All calls to GetChecksum come after Finalize
46 class FileChecksumGenerator {
47  public:
48  virtual ~FileChecksumGenerator() {}
49 
50  // Update the current result after process the data. For different checksum
51  // functions, the temporal results may be stored and used in Update to
52  // include the new data.
53  virtual void Update(const char* data, size_t n) = 0;
54 
55  // Generate the final results if no further new data will be updated.
56  virtual void Finalize() = 0;
57 
58  // Get the checksum. The result should not be the empty string and may
59  // include arbitrary bytes, including non-printable characters.
60  virtual std::string GetChecksum() const = 0;
61 
62  // Returns a name that identifies the current file checksum function.
63  virtual const char* Name() const = 0;
64 };
