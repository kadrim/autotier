31 // A BlockBasedFilterBlockBuilder is used to construct all of the filters for a
32 // particular Table. It generates a single string which is stored as
33 // a special block in the Table.
34 //
35 // The sequence of calls to BlockBasedFilterBlockBuilder must match the regexp:
36 // (StartBlock Add*)* Finish
37 class BlockBasedFilterBlockBuilder : public FilterBlockBuilder {
38  public:
39  BlockBasedFilterBlockBuilder(const SliceTransform* prefix_extractor,
40  const BlockBasedTableOptions& table_opt);
41  // No copying allowed
42  BlockBasedFilterBlockBuilder(const BlockBasedFilterBlockBuilder&) = delete;
43  void operator=(const BlockBasedFilterBlockBuilder&) = delete;
44 
45  virtual bool IsBlockBased() override { return true; }
46  virtual void StartBlock(uint64_t block_offset) override;
47  virtual void Add(const Slice& key_without_ts) override;
48  virtual bool IsEmpty() const override {
49  return start_.empty() && filter_offsets_.empty();
50  }
51  virtual size_t EstimateEntriesAdded() override;
52  virtual Slice Finish(const BlockHandle& tmp, Status* status) override;
53  using FilterBlockBuilder::Finish;
54 
55  private:
56  void AddKey(const Slice& key);
57  void AddPrefix(const Slice& key);
58  void GenerateFilter();
59 
60  // important: all of these might point to invalid addresses
61  // at the time of destruction of this filter block. destructor
62  // should NOT dereference them.
63  const FilterPolicy* policy_;
64  const SliceTransform* prefix_extractor_;
65  bool whole_key_filtering_;
66 
67  size_t prev_prefix_start_; // the position of the last appended prefix
68  // to "entries_".
69  size_t prev_prefix_size_; // the length of the last appended prefix to
70  // "entries_".
71  std::string entries_; // Flattened entry contents
72  std::vector<size_t> start_; // Starting index in entries_ of each entry
73  std::string result_; // Filter data computed so far
74  std::vector<Slice> tmp_entries_; // policy_->CreateFilter() argument
75  std::vector<uint32_t> filter_offsets_;
76  uint64_t total_added_in_built_; // Total keys added to filters built so far
77 };
