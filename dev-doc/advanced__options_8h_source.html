40 // In Level-based compaction, it Determines which file from a level to be
41 // picked to merge to the next level. We suggest people try
42 // kMinOverlappingRatio first when you tune your database.
43 enum CompactionPri : char {
44  // Slightly prioritize larger files by size compensated by #deletes
45  kByCompensatedSize = 0x0,
46  // First compact files whose data's latest update time is oldest.
47  // Try this if you only update some hot keys in small ranges.
48  kOldestLargestSeqFirst = 0x1,
49  // First compact files whose range hasn't been compacted to the next level
50  // for the longest. If your updates are random across the key space,
51  // write amplification is slightly better with this option.

54  // and its size is the smallest. It in many cases can optimize write
55  // amplification.
56  kMinOverlappingRatio = 0x3,

89  // The default value is the max 16-bit int as it'll be written out in OPTIONS

97  // Maximum size of dictionaries used to prime the compression library.
98  // Enabling dictionary can improve compression ratios when there are
99  // repetitions across data blocks.
100  //
101  // The dictionary is created by sampling the SST file data. If
102  // `zstd_max_train_bytes` is nonzero, the samples are passed through zstd's
103  // dictionary generator. Otherwise, the random samples are used directly as
104  // the dictionary.
105  //
106  // When compression dictionary is disabled, we compress and write each block
107  // before buffering data for the next one. When compression dictionary is
108  // enabled, we buffer all SST file data in-memory so we can sample it, as data
109  // can only be compressed and written after the dictionary has been finalized.
110  // So users of this feature may see increased memory usage.
111  //
112  // Default: 0.
113  uint32_t max_dict_bytes;
114 
115  // Maximum size of training data passed to zstd's dictionary trainer. Using
116  // zstd's dictionary trainer can achieve even better compression ratio
117  // improvements than using `max_dict_bytes` alone.
118  //
119  // The training data will be used to generate a dictionary of max_dict_bytes.
120  //
121  // Default: 0.
122  uint32_t zstd_max_train_bytes;
123 
124  // Number of threads for parallel compression.
125  // Parallel compression is enabled only if threads > 1.
126  // THE FEATURE IS STILL EXPERIMENTAL
127  //
128  // This option is valid only when BlockBasedTable is used.
129  //
130  // When parallel compression is enabled, SST size file sizes might be
131  // more inflated compared to the target size, because more data of unknown
132  // compressed size is in flight when compression is parallelized. To be
133  // reasonably accurate, this inflation is also estimated by using historical
134  // compression ratio and current bytes inflight.
135  //
136  // Default: 1.
137  uint32_t parallel_threads;
138 
139  // When the compression options are set by the user, it will be set to "true".
140  // For bottommost_compression_opts, to enable it, user must set enabled=true.
141  // Otherwise, bottommost compression will use compression_opts as default
142  // compression options.
143  //
144  // For compression_opts, if compression_opts.enabled=false, it is still
145  // used as compression options for compression process.
146  //
147  // Default: false.
148  bool enabled;
149 
150  // Limit on data buffering when gathering samples to build a dictionary. Zero
151  // means no limit. When dictionary is disabled (`max_dict_bytes == 0`),
152  // enabling this limit (`max_dict_buffer_bytes != 0`) has no effect.
153  //
154  // In compaction, the buffering is limited to the target file size (see
155  // `target_file_size_base` and `target_file_size_multiplier`) even if this
156  // setting permits more buffering. Since we cannot determine where the file
157  // should be cut until data blocks are compressed with dictionary, buffering
158  // more than the target file size could lead to selecting samples that belong
159  // to a later output SST.
160  //
161  // Limiting too strictly may harm dictionary effectiveness since it forces
162  // RocksDB to pick samples from the initial portion of the output SST, which
163  // may not be representative of the whole file. Configuring this limit below
164  // `zstd_max_train_bytes` (when enabled) can restrict how many samples we can
165  // pass to the dictionary trainer. Configuring it below `max_dict_bytes` can
166  // restrict the size of the final dictionary.
167  //
168  // Default: 0 (unlimited)
169  uint64_t max_dict_buffer_bytes;

219 struct AdvancedColumnFamilyOptions {
220  // The maximum number of write buffers that are built up in memory.
221  // The default and the minimum number is 2, so that when 1 write buffer
222  // is being flushed to storage, new writes can continue to the other
223  // write buffer.
224  // If max_write_buffer_number > 3, writing will be slowed down to
225  // options.delayed_write_rate if we are writing to the last write buffer
226  // allowed.
227  //
228  // Default: 2
229  //
230  // Dynamically changeable through SetOptions() API
231  int max_write_buffer_number = 2;
232 
233  // The minimum number of write buffers that will be merged together
234  // before writing to storage. If set to 1, then
235  // all write buffers are flushed to L0 as individual files and this increases
236  // read amplification because a get request has to check in all of these
237  // files. Also, an in-memory merge may result in writing lesser
238  // data to storage if there are duplicate records in each of these
239  // individual write buffers. Default: 1
240  int min_write_buffer_number_to_merge = 1;
241 
242  // DEPRECATED
243  // The total maximum number of write buffers to maintain in memory including
244  // copies of buffers that have already been flushed. Unlike
245  // max_write_buffer_number, this parameter does not affect flushing.
246  // This parameter is being replaced by max_write_buffer_size_to_maintain.
247  // If both parameters are set to non-zero values, this parameter will be
248  // ignored.
249  int max_write_buffer_number_to_maintain = 0;
250 
251  // The total maximum size(bytes) of write buffers to maintain in memory
252  // including copies of buffers that have already been flushed. This parameter
253  // only affects trimming of flushed buffers and does not affect flushing.
254  // This controls the maximum amount of write history that will be available
255  // in memory for conflict checking when Transactions are used. The actual
256  // size of write history (flushed Memtables) might be higher than this limit
257  // if further trimming will reduce write history total size below this
258  // limit. For example, if max_write_buffer_size_to_maintain is set to 64MB,
259  // and there are three flushed Memtables, with sizes of 32MB, 20MB, 20MB.
260  // Because trimming the next Memtable of size 20MB will reduce total memory
261  // usage to 52MB which is below the limit, RocksDB will stop trimming.
262  //
263  // When using an OptimisticTransactionDB:
264  // If this value is too low, some transactions may fail at commit time due
265  // to not being able to determine whether there were any write conflicts.
266  //
267  // When using a TransactionDB:
268  // If Transaction::SetSnapshot is used, TransactionDB will read either
269  // in-memory write buffers or SST files to do write-conflict checking.
270  // Increasing this value can reduce the number of reads to SST files
271  // done for conflict detection.
272  //
273  // Setting this value to 0 will cause write buffers to be freed immediately
274  // after they are flushed. If this value is set to -1,
275  // 'max_write_buffer_number * write_buffer_size' will be used.
276  //
277  // Default:
278  // If using a TransactionDB/OptimisticTransactionDB, the default value will
279  // be set to the value of 'max_write_buffer_number * write_buffer_size'
280  // if it is not explicitly set by the user. Otherwise, the default is 0.
281  int64_t max_write_buffer_size_to_maintain = 0;

283  // Allows thread-safe inplace updates. If this is true, there is no way to
284  // achieve point-in-time consistency using snapshot or iterator (assuming
285  // concurrent updates). Hence iterator and multi-get will return results
286  // which are not consistent as of any point-in-time.
287  // Backward iteration on memtables will not work either.
288  // If inplace_callback function is not set,
289  // Put(key, new_value) will update inplace the existing_value iff
290  // * key exists in current memtable
291  // * new sizeof(new_value) <= sizeof(existing_value)
292  // * existing_value for that key is a put i.e. kTypeValue
293  // If inplace_callback function is set, check doc for inplace_callback.
294  // Default: false.
295  bool inplace_update_support = false;

297  // Number of locks used for inplace update
298  // Default: 10000, if inplace_update_support = true, else 0.
299  //
300  // Dynamically changeable through SetOptions() API
301  size_t inplace_update_num_locks = 10000;

303  // existing_value - pointer to previous value (from both memtable and sst).
304  // nullptr if key doesn't exist
305  // existing_value_size - pointer to size of existing_value).
306  // nullptr if key doesn't exist
307  // delta_value - Delta value to be merged with the existing_value.
308  // Stored in transaction logs.
309  // merged_value - Set when delta is applied on the previous value.
310 
311  // Applicable only when inplace_update_support is true,
312  // this callback function is called at the time of updating the memtable
313  // as part of a Put operation, lets say Put(key, delta_value). It allows the
314  // 'delta_value' specified as part of the Put operation to be merged with
315  // an 'existing_value' of the key in the database.
316 
317  // If the merged value is smaller in size that the 'existing_value',
318  // then this function can update the 'existing_value' buffer inplace and
319  // the corresponding 'existing_value'_size pointer, if it wishes to.
320  // The callback should return UpdateStatus::UPDATED_INPLACE.
321  // In this case. (In this case, the snapshot-semantics of the rocksdb
322  // Iterator is not atomic anymore).
323

330

423  // Different levels can have different compression policies. There
424  // are cases where most lower levels would like to use quick compression
425  // algorithms while the higher levels (which have more data) use
426  // compression algorithms that have better compression but could

465  // Target file size for compaction.
466  // target_file_size_base is per-file size for level-1.
467  // Target file size for level L can be calculated by
468  // target_file_size_base * (target_file_size_multiplier ^ (L-1))
469  // For example, if target_file_size_base is 2MB and
470  // target_file_size_multiplier is 10, then each file on level-1 will
471  // be 2MB, and each file on level 2 will be 20MB,
472  // and each file on level-3 will be 200MB.
473  //
474  // Default: 64MB.
475  //
476  // Dynamically changeable through SetOptions() API
477  uint64_t target_file_size_base = 64 * 1048576;

479  // By default target_file_size_multiplier is 1, which means
480  // by default files in different levels will have similar size.
481  //
482  // Dynamically changeable through SetOptions() API
483  int target_file_size_multiplier = 1;
484 
485  // If true, RocksDB will pick target size of each level dynamically.
486  // We will pick a base level b >= 1. L0 will be directly merged into level b,
487  // instead of always into level 1. Level 1 to b-1 need to be empty.
488  // We try to pick b and its target size so that
489  // 1. target size is in the range of
490  // (max_bytes_for_level_base / max_bytes_for_level_multiplier,
491  // max_bytes_for_level_base]
492  // 2. target size of the last level (level num_levels-1) equals to extra size
493  // of the level.
494  // At the same time max_bytes_for_level_multiplier and
495  // max_bytes_for_level_multiplier_additional are still satisfied.
496  // (When L0 is too large, we make some adjustment. See below.)
497  //
498  // With this option on, from an empty DB, we make last level the base level,
499  // which means merging L0 data into the last level, until it exceeds
500  // max_bytes_for_level_base. And then we make the second last level to be
501  // base level, to start to merge L0 data to second last level, with its
502  // target size to be 1/max_bytes_for_level_multiplier of the last level's
503  // extra size. After the data accumulates more so that we need to move the
504  // base level to the third last one, and so on.
505  //
506  // For example, assume max_bytes_for_level_multiplier=10, num_levels=6,
507  // and max_bytes_for_level_base=10MB.
508  // Target sizes of level 1 to 5 starts with:
509  // [- - - - 10MB]
510  // with base level is level. Target sizes of level 1 to 4 are not applicable
511  // because they will not be used.
512  // Until the size of Level 5 grows to more than 10MB, say 11MB, we make
513  // base target to level 4 and now the targets looks like:
514  // [- - - 1.1MB 11MB]
515  // While data are accumulated, size targets are tuned based on actual data
516  // of level 5. When level 5 has 50MB of data, the target is like:
517  // [- - - 5MB 50MB]
518  // Until level 5's actual size is more than 100MB, say 101MB. Now if we keep
519  // level 4 to be the base level, its target size needs to be 10.1MB, which
520  // doesn't satisfy the target size range. So now we make level 3 the target
521  // size and the target sizes of the levels look like:
522  // [- - 1.01MB 10.1MB 101MB]
523  // In the same way, while level 5 further grows, all levels' targets grow,
524  // like
525  // [- - 5MB 50MB 500MB]
526  // Until level 5 exceeds 1000MB and becomes 1001MB, we make level 2 the
527  // base level and make levels' target sizes like this:
528  // [- 1.001MB 10.01MB 100.1MB 1001MB]
529  // and go on...
530  //
531  // By doing it, we give max_bytes_for_level_multiplier a priority against
532  // max_bytes_for_level_base, for a more predictable LSM tree shape. It is
533  // useful to limit worse case space amplification.
534  //
535  //
536  // If the compaction from L0 is lagged behind, a special mode will be turned
537  // on to prioritize write amplification against max_bytes_for_level_multiplier
538  // or max_bytes_for_level_base. The L0 compaction is lagged behind by looking
539  // at number of L0 files and total L0 size. If number of L0 files is at least
540  // the double of level0_file_num_compaction_trigger, or the total size is
541  // at least max_bytes_for_level_base, this mode is on. The target of L1 grows
542  // to the actual data size in L0, and then determine the target for each level
543  // so that each level will have the same level multiplier.
544  //
545  // For example, when L0 size is 100MB, the size of last level is 1600MB,
546  // max_bytes_for_level_base = 80MB, and max_bytes_for_level_multiplier = 10.
547  // Since L0 size is larger than max_bytes_for_level_base, this is a L0
548  // compaction backlogged mode. So that the L1 size is determined to be 100MB.
549  // Based on max_bytes_for_level_multiplier = 10, at least 3 non-0 levels will
550  // be needed. The level multiplier will be calculated to be 4 and the three
