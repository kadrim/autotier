64 // a key range buffer represents a set of key ranges that can
65 // be stored, iterated over, and then destroyed all at once.
66 class range_buffer {
67  private:
68  // the key range buffer is a bunch of records in a row.
69  // each record has the following header, followed by the
70  // left key and right key data payload, if applicable.
71  // we limit keys to be 2^16, since we store lengths as 2 bytes.
72  static const size_t MAX_KEY_SIZE = 1 << 16;
73 
74  struct record_header {
75  bool left_neg_inf;
76  bool left_pos_inf;
77  bool right_pos_inf;
78  bool right_neg_inf;
79  uint16_t left_key_size;
80  uint16_t right_key_size;
81  bool is_exclusive_lock;
82 
83  bool left_is_infinite(void) const;
84 
85  bool right_is_infinite(void) const;
86 
87  void init(const DBT *left_key, const DBT *right_key, bool is_exclusive);
88  };
89  // PORT static_assert(sizeof(record_header) == 8, "record header format is
90  // off");
91 
92  public:
93  // the iterator abstracts reading over a buffer of variable length
94  // records one by one until there are no more left.
95  class iterator {
96  public:
97  iterator();
98  iterator(const range_buffer *buffer);
99 
100  // a record represents the user-view of a serialized key range.
101  // it handles positive and negative infinity and the optimized
102  // point range case, where left and right points share memory.
103  class record {
104  public:
105  // get a read-only pointer to the left key of this record's range
106  const DBT *get_left_key(void) const;
107 
108  // get a read-only pointer to the right key of this record's range
109  const DBT *get_right_key(void) const;
110 
111  // how big is this record? this tells us where the next record is
112  size_t size(void) const;
113 
114  bool get_exclusive_flag() const { return _header.is_exclusive_lock; }
115 
116  // populate a record header and point our DBT's
117  // buffers into ours if they are not infinite.
118  void deserialize(const char *buf);
119 
120  private:
121  record_header _header;
122  DBT _left_key;
123  DBT _right_key;
124  };
125 
126  // populate the given record object with the current
127  // the memory referred to by record is valid for only
128  // as long as the record exists.
129  bool current(record *rec);
130 
131  // move the iterator to the next record in the buffer
132  void next(void);
133 
134  private:
135  void reset_current_chunk();
136 
137  // the key range buffer we are iterating over, the current
138  // offset in that buffer, and the size of the current record.
139  memarena::chunk_iterator _ma_chunk_iterator;
140  const void *_current_chunk_base;
141  size_t _current_chunk_offset;
142  size_t _current_chunk_max;
143  size_t _current_rec_size;
144  };
145 
146  // allocate buffer space lazily instead of on creation. this way,
147  // no malloc/free is done if the transaction ends up taking no locks.
148  void create(void);
149 
150  // append a left/right key range to the buffer.
151  // if the keys are equal, then only one copy is stored.
152  void append(const DBT *left_key, const DBT *right_key,
153  bool is_write_request = false);
154 
155  // is this range buffer empty?
156  bool is_empty(void) const;
157 
158  // how much memory is being used by this range buffer?
159  uint64_t total_memory_size(void) const;
160 
161  // how many ranges are stored in this range buffer?
162  int get_num_ranges(void) const;
163 
164  void destroy(void);
165 
166  private:
167  memarena _arena;
168  int _num_ranges;
169 
170  void append_range(const DBT *left_key, const DBT *right_key,
171  bool is_write_request);
172 
173  // append a point to the buffer. this is the space/time saving
174  // optimization for key ranges where left == right.
175  void append_point(const DBT *key, bool is_write_request);
176 };
