28 // BlockAccessCipherStream is the base class for any cipher stream that
29 // supports random access at block level (without requiring data from other
30 // blocks). E.g. CTR (Counter operation mode) supports this requirement.
31 class BlockAccessCipherStream {
32  public:
33  virtual ~BlockAccessCipherStream(){};
34 
35  // BlockSize returns the size of each block supported by this cipher stream.
36  virtual size_t BlockSize() = 0;
37 
38  // Encrypt one or more (partial) blocks of data at the file offset.
39  // Length of data is given in dataSize.
40  virtual Status Encrypt(uint64_t fileOffset, char* data, size_t dataSize);
41 
42  // Decrypt one or more (partial) blocks of data at the file offset.
43  // Length of data is given in dataSize.
44  virtual Status Decrypt(uint64_t fileOffset, char* data, size_t dataSize);
45 
46  protected:
47  // Allocate scratch space which is passed to EncryptBlock/DecryptBlock.
48  virtual void AllocateScratch(std::string&) = 0;
49 
50  // Encrypt a block of data at the given block index.
51  // Length of data is equal to BlockSize();
52  virtual Status EncryptBlock(uint64_t blockIndex, char* data,
53  char* scratch) = 0;
54 
55  // Decrypt a block of data at the given block index.
56  // Length of data is equal to BlockSize();
57  virtual Status DecryptBlock(uint64_t blockIndex, char* data,
58  char* scratch) = 0;
