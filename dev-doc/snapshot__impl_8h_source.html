20 // Snapshots are kept in a doubly-linked list in the DB.
21 // Each SnapshotImpl corresponds to a particular sequence number.
22 class SnapshotImpl : public Snapshot {
23  public:
24  SequenceNumber number_; // const after creation
25  // It indicates the smallest uncommitted data at the time the snapshot was
26  // taken. This is currently used by WritePrepared transactions to limit the
27  // scope of queries to IsInSnapshot.
28  SequenceNumber min_uncommitted_ = kMinUnCommittedSeq;
29 
30  virtual SequenceNumber GetSequenceNumber() const override { return number_; }
31 
32  private:
33  friend class SnapshotList;
34 
35  // SnapshotImpl is kept in a doubly-linked circular list
36  SnapshotImpl* prev_;
37  SnapshotImpl* next_;
38 
39  SnapshotList* list_; // just for sanity checks
40 
41  int64_t unix_time_;
42 
43  // Will this snapshot be used by a Transaction to do write-conflict checking?
44  bool is_write_conflict_boundary_;
45 };

47 class SnapshotList {
48  public:
49  SnapshotList() {
50  list_.prev_ = &list_;
51  list_.next_ = &list_;
52  list_.number_ = 0xFFFFFFFFL; // placeholder marker, for debugging
53  // Set all the variables to make UBSAN happy.
54  list_.list_ = nullptr;
55  list_.unix_time_ = 0;
56  list_.is_write_conflict_boundary_ = false;
57  count_ = 0;

60  // No copy-construct.
61  SnapshotList(const SnapshotList&) = delete;

63  bool empty() const { return list_.next_ == &list_; }
64  SnapshotImpl* oldest() const { assert(!empty()); return list_.next_; }
65  SnapshotImpl* newest() const { assert(!empty()); return list_.prev_; }

67  SnapshotImpl* New(SnapshotImpl* s, SequenceNumber seq, uint64_t unix_time,
68  bool is_write_conflict_boundary) {
69  s->number_ = seq;
70  s->unix_time_ = unix_time;
71  s->is_write_conflict_boundary_ = is_write_conflict_boundary;
72  s->list_ = this;
73  s->next_ = &list_;
74  s->prev_ = list_.prev_;
75  s->prev_->next_ = s;
76  s->next_->prev_ = s;
77  count_++;
78  return s;
79  }

81  // Do not responsible to free the object.
82  void Delete(const SnapshotImpl* s) {
83  assert(s->list_ == this);
84  s->prev_->next_ = s->next_;
85  s->next_->prev_ = s->prev_;
86  count_--;
87  }

99  void GetAll(std::vector<SequenceNumber>* snap_vector,
100  SequenceNumber* oldest_write_conflict_snapshot = nullptr,
101  const SequenceNumber& max_seq = kMaxSequenceNumber) const {
102  std::vector<SequenceNumber>& ret = *snap_vector;
103  // So far we have no use case that would pass a non-empty vector
104  assert(ret.size() == 0);
105 
106  if (oldest_write_conflict_snapshot != nullptr) {
107  *oldest_write_conflict_snapshot = kMaxSequenceNumber;
108  }
109 
110  if (empty()) {
111  return;
112  }
113  const SnapshotImpl* s = &list_;
114  while (s->next_ != &list_) {
115  if (s->next_->number_ > max_seq) {
116  break;
117  }
118  // Avoid duplicates
119  if (ret.empty() || ret.back() != s->next_->number_) {
120  ret.push_back(s->next_->number_);
121  }
122 
123  if (oldest_write_conflict_snapshot != nullptr &&
124  *oldest_write_conflict_snapshot == kMaxSequenceNumber &&
125  s->next_->is_write_conflict_boundary_) {
126  // If this is the first write-conflict boundary snapshot in the list,
127  // it is the oldest
128  *oldest_write_conflict_snapshot = s->next_->number_;
129  }
130 
131  s = s->next_;
132  }
133  return;
134  }
