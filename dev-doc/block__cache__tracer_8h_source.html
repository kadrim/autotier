49 // Lookup context for tracing block cache accesses.
50 // We trace block accesses at five places:
51 // 1. BlockBasedTable::GetFilter
52 // 2. BlockBasedTable::GetUncompressedDict.
53 // 3. BlockBasedTable::MaybeReadAndLoadToCache. (To trace access on data, index,
54 // and range deletion block.)
55 // 4. BlockBasedTable::Get. (To trace the referenced key and whether the
56 // referenced key exists in a fetched data block.)
57 // 5. BlockBasedTable::MultiGet. (To trace the referenced key and whether the
58 // referenced key exists in a fetched data block.)
59 // The context is created at:
60 // 1. BlockBasedTable::Get. (kUserGet)
61 // 2. BlockBasedTable::MultiGet. (kUserMGet)
62 // 3. BlockBasedTable::NewIterator. (either kUserIterator, kCompaction, or
63 // external SST ingestion calls this function.)
64 // 4. BlockBasedTable::Open. (kPrefetch)
65 // 5. Index/Filter::CacheDependencies. (kPrefetch)
66 // 6. BlockBasedTable::ApproximateOffsetOf. (kCompaction or
67 // kUserApproximateSize).
68 struct BlockCacheLookupContext {
69  BlockCacheLookupContext(const TableReaderCaller& _caller) : caller(_caller) {}
70  BlockCacheLookupContext(const TableReaderCaller& _caller, uint64_t _get_id,
71  bool _get_from_user_specified_snapshot)
72  : caller(_caller),
73  get_id(_get_id),
74  get_from_user_specified_snapshot(_get_from_user_specified_snapshot) {}
75  const TableReaderCaller caller;
76  // These are populated when we perform lookup/insert on block cache. The block
77  // cache tracer uses these inforation when logging the block access at
78  // BlockBasedTable::GET and BlockBasedTable::MultiGet.
79  bool is_cache_hit = false;
80  bool no_insert = false;
81  TraceType block_type = TraceType::kTraceMax;
82  uint64_t block_size = 0;
83  std::string block_key;
84  uint64_t num_keys_in_block = 0;
85  // The unique id associated with Get and MultiGet. This enables us to track
86  // how many blocks a Get/MultiGet request accesses. We can also measure the
87  // impact of row cache vs block cache.
88  uint64_t get_id = 0;
89  std::string referenced_key;
90  bool get_from_user_specified_snapshot = false;
