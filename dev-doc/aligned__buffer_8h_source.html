18 // Truncate to a multiple of page_size, which is also a page boundary. This
19 // helps to figuring out the right alignment.
20 // Example:
21 // TruncateToPageBoundary(4096, 5000) => 4096
22 // TruncateToPageBoundary((4096, 10000) => 8192
23 inline size_t TruncateToPageBoundary(size_t page_size, size_t s) {
24  s -= (s & (page_size - 1));
25  assert((s % page_size) == 0);
26  return s;
27 }

29 // Round up x to a multiple of y.
30 // Example:
31 // Roundup(13, 5) => 15
32 // Roundup(201, 16) => 208
33 inline size_t Roundup(size_t x, size_t y) {
34  return ((x + y - 1) / y) * y;
35 }

37 // Round down x to a multiple of y.
38 // Example:
39 // Rounddown(13, 5) => 10
40 // Rounddown(201, 16) => 192
41 inline size_t Rounddown(size_t x, size_t y) { return (x / y) * y; }

43 // AlignedBuffer manages a buffer by taking alignment into consideration, and
44 // aligns the buffer start and end positions. It is mainly used for direct I/O,
45 // though it can be used other purposes as well.
46 // It also supports expanding the managed buffer, and copying whole or part of
47 // the data from old buffer into the new expanded buffer. Such a copy especially
48 // helps in cases avoiding an IO to re-fetch the data from disk.
49 //
50 // Example:
51 // AlignedBuffer buf;
52 // buf.Alignment(alignment);
53 // buf.AllocateNewBuffer(user_requested_buf_size);
54 // ...
55 // buf.AllocateNewBuffer(2*user_requested_buf_size, /*copy_data*/ true,
56 // copy_offset, copy_len);

132  // Allocates a new buffer and sets the start position to the first aligned
133  // byte.
134  //
135  // requested_capacity: requested new buffer capacity. This capacity will be
136  // rounded up based on alignment.
137  // copy_data: Copy data from old buffer to new buffer. If copy_offset and
138  // copy_len are not passed in and the new requested capacity is bigger
139  // than the existing buffer's capacity, the data in the exising buffer is
140  // fully copied over to the new buffer.
141  // copy_offset: Copy data from this offset in old buffer.
142  // copy_len: Number of bytes to copy.
143  //
144  // The function does nothing if the new requested_capacity is smaller than
145  // the current buffer capacity and copy_data is true i.e. the old buffer is
146  // retained as is.
147  void AllocateNewBuffer(size_t requested_capacity, bool copy_data = false,
148  uint64_t copy_offset = 0, size_t copy_len = 0) {
149  assert(alignment_ > 0);
150  assert((alignment_ & (alignment_ - 1)) == 0);
151 
152  copy_len = copy_len > 0 ? copy_len : cursize_;
153  if (copy_data && requested_capacity < copy_len) {
154  // If we are downsizing to a capacity that is smaller than the current
155  // data in the buffer -- Ignore the request.
156  return;
157  }
158 
159  size_t new_capacity = Roundup(requested_capacity, alignment_);
160  char* new_buf = new char[new_capacity + alignment_];
161  char* new_bufstart = reinterpret_cast<char*>(
162  (reinterpret_cast<uintptr_t>(new_buf) + (alignment_ - 1)) &
163  ~static_cast<uintptr_t>(alignment_ - 1));
164 
165  if (copy_data) {
166  assert(bufstart_ + copy_offset + copy_len <= bufstart_ + cursize_);
167  memcpy(new_bufstart, bufstart_ + copy_offset, copy_len);
168  cursize_ = copy_len;
169  } else {
170  cursize_ = 0;
171  }
172 
173  bufstart_ = new_bufstart;
174  capacity_ = new_capacity;
175  buf_.reset(new_buf);
176  }
177 
178  // Append to the buffer.
179  //
180  // src : source to copy the data from.
181  // append_size : number of bytes to copy from src.
182  // Returns the number of bytes appended.
183  //
184  // If append_size is more than the remaining buffer size only the
185  // remaining-size worth of bytes are copied.
186  size_t Append(const char* src, size_t append_size) {
187  size_t buffer_remaining = capacity_ - cursize_;
188  size_t to_copy = std::min(append_size, buffer_remaining);
189 
190  if (to_copy > 0) {
191  memcpy(bufstart_ + cursize_, src, to_copy);
192  cursize_ += to_copy;
193  }
194  return to_copy;
195  }
196 
197  // Read from the buffer.
198  //
199  // dest : destination buffer to copy the data to.
200  // offset : the buffer offset to start reading from.
201  // read_size : the number of bytes to copy from the buffer to dest.
202  // Returns the number of bytes read/copied to dest.
203  size_t Read(char* dest, size_t offset, size_t read_size) const {
204  assert(offset < cursize_);
205 
206  size_t to_read = 0;
207  if(offset < cursize_) {
208  to_read = std::min(cursize_ - offset, read_size);
209  }
210  if (to_read > 0) {
211  memcpy(dest, bufstart_ + offset, to_read);
212  }
213  return to_read;
214  }

216  // Pad to the end of alignment with "padding"
217  void PadToAlignmentWith(int padding) {
218  size_t total_size = Roundup(cursize_, alignment_);
219  size_t pad_size = total_size - cursize_;
220 
221  if (pad_size > 0) {
222  assert((pad_size + cursize_) <= capacity_);
223  memset(bufstart_ + cursize_, padding, pad_size);
224  cursize_ += pad_size;
225  }
226  }
227 
228  void PadWith(size_t pad_size, int padding) {
229  assert((pad_size + cursize_) <= capacity_);
230  memset(bufstart_ + cursize_, padding, pad_size);
231  cursize_ += pad_size;
232  }

234  // After a partial flush move the tail to the beginning of the buffer.
235  void RefitTail(size_t tail_offset, size_t tail_size) {
236  if (tail_size > 0) {
237  memmove(bufstart_, bufstart_ + tail_offset, tail_size);
238  }
