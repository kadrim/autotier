21 // StatsHistoryIterator is the main interface for users to programmatically
22 // access statistics snapshots that was automatically stored by RocksDB.
23 // Depending on options, the stats can be in memory or on disk.
24 // The stats snapshots are indexed by time that they were recorded, and each
25 // stats snapshot contains individual stat name and value at the time of
26 // recording.
27 // Example:
28 // std::unique_ptr<StatsHistoryIterator> stats_iter;
29 // Status s = db->GetStatsHistory(0 /* start_time */,
30 // env->NowMicros() /* end_time*/,
31 // &stats_iter);
32 // if (s.ok) {
33 // for (; stats_iter->Valid(); stats_iter->Next()) {
34 // uint64_t stats_time = stats_iter->GetStatsTime();
35 // const std::map<std::string, uint64_t>& stats_map =
36 // stats_iter->GetStatsMap();
37 // process(stats_time, stats_map);
38 // }
39 // }
40 class StatsHistoryIterator {
41  public:
42  StatsHistoryIterator() {}
43  virtual ~StatsHistoryIterator() {}
44 
45  virtual bool Valid() const = 0;
46 
47  // Moves to the next stats history record. After this call, Valid() is
48  // true iff the iterator was not positioned at the last entry in the source.
49  // REQUIRES: Valid()
50  virtual void Next() = 0;
51 
52  // Return the time stamp (in seconds) when stats history is recorded.
53  // REQUIRES: Valid()
54  virtual uint64_t GetStatsTime() const = 0;
55 
56  virtual int GetFormatVersion() const { return -1; }
57 
58  // Return the current stats history as an std::map which specifies the
59  // mapping from stats name to stats value . The underlying storage
60  // for the returned map is valid only until the next modification of
61  // the iterator.
62  // REQUIRES: Valid()
63  virtual const std::map<std::string, uint64_t>& GetStatsMap() const = 0;
64 
65  // If an error has occurred, return it. Else return an ok status.
66  virtual Status status() const = 0;
67 };
