77  // In mmaped mode, the results point to mmaped area of the file, which
78  // means it is always valid before closing the file.
79  // In non-mmap mode, the results point to an internal buffer. If the caller
80  // makes another read call, the results may not be valid. So callers should
81  // make a copy when needed.
82  // In order to save read calls to files, we keep two internal buffers:
83  // the first read and the most recent read. This is efficient because it
84  // columns these two common use cases:
85  // (1) hash index only identify one location, we read the key to verify
86  // the location, and read key and value if it is the right location.
87  // (2) after hash index checking, we identify two locations (because of
88  // hash bucket conflicts), we binary search the two location to see
89  // which one is what we need and start to read from the location.
90  // These two most common use cases will be covered by the two buffers
91  // so that we don't need to re-read the same location.
92  // Currently we keep a fixed size buffer. If a read doesn't exactly fit
93  // the buffer, we replace the second buffer with the location user reads.
94  //
95  // If return false, status code is stored in status_.
96  bool Read(uint32_t file_offset, uint32_t len, Slice* out) {
97  if (file_info_->is_mmap_mode) {
98  assert(file_offset + len <= file_info_->data_end_offset);
99  *out = Slice(file_info_->file_data.data() + file_offset, len);
100  return true;
101  } else {
102  return ReadNonMmap(file_offset, len, out);
103  }
104  }
105 
106  // If return false, status code is stored in status_.
107  bool ReadNonMmap(uint32_t file_offset, uint32_t len, Slice* output);
