68  // The table file creation process invokes this method before adding a kv to
69  // the table file. A return value of false indicates that the kv should be
70  // preserved in the new table file and a return value of true indicates
71  // that this key-value should be removed from the new table file. The
72  // application can inspect the existing value of the key and make decision
73  // based on it.
74  //
75  // Key-Values that are results of merge operation during table file creation
76  // are not passed into this function. Currently, when you have a mix of Put()s
77  // and Merge()s on a same key, we only guarantee to process the merge operands
78  // through the `CompactionFilter`s. Put()s might be processed, or might not.
79  //
80  // When the value is to be preserved, the application has the option
81  // to modify the existing_value and pass it back through new_value.
82  // value_changed needs to be set to true in this case.
83  //
84  // Note that RocksDB snapshots (i.e. call GetSnapshot() API on a
85  // DB* object) will not guarantee to preserve the state of the DB with
86  // CompactionFilter. Data seen from a snapshot might disappear after a
87  // table file created with a `CompactionFilter` is installed. If you use
88  // snapshots, think twice about whether you want to use `CompactionFilter` and
89  // whether you are using it in a safe way.
90  //
91  // If multithreaded compaction is being used *and* a single CompactionFilter
92  // instance was supplied via Options::compaction_filter, this method may be
93  // called from different threads concurrently. The application must ensure
94  // that the call is thread-safe.
95  //
96  // If the CompactionFilter was created by a factory, then it will only ever
97  // be used by a single thread that is doing the table file creation, and this
98  // call does not need to be thread-safe. However, multiple filters may be
99  // in existence and operating concurrently.
100  virtual bool Filter(int /*level*/, const Slice& /*key*/,
101  const Slice& /*existing_value*/,
102  std::string* /*new_value*/,
103  bool* /*value_changed*/) const {
104  return false;
105  }
106 
107  // The table file creation process invokes this method on every merge operand.
108  // If this method returns true, the merge operand will be ignored and not
109  // written out in the new table file.
110  //
111  // Note: If you are using a TransactionDB, it is not recommended to implement
112  // FilterMergeOperand(). If a Merge operation is filtered out, TransactionDB
113  // may not realize there is a write conflict and may allow a Transaction to
114  // Commit that should have failed. Instead, it is better to implement any
115  // Merge filtering inside the MergeOperator.
116  virtual bool FilterMergeOperand(int /*level*/, const Slice& /*key*/,
117  const Slice& /*operand*/) const {
118  return false;
119  }
120 
121  // An extended API. Called for both values and merge operands.
122  // Allows changing value and skipping ranges of keys.
123  // The default implementation uses Filter() and FilterMergeOperand().
124  // If you're overriding this method, no need to override the other two.
125  // `value_type` indicates whether this key-value corresponds to a normal
126  // value (e.g. written with Put()) or a merge operand (written with Merge()).
127  //
128  // Possible return values:
129  // * kKeep - keep the key-value pair.
130  // * kRemove - remove the key-value pair or merge operand.
131  // * kChangeValue - keep the key and change the value/operand to *new_value.
132  // * kRemoveAndSkipUntil - remove this key-value pair, and also remove
133  // all key-value pairs with key in [key, *skip_until). This range
134  // of keys will be skipped without reading, potentially saving some
135  // IO operations compared to removing the keys one by one.
136  //
137  // *skip_until <= key is treated the same as Decision::kKeep
138  // (since the range [key, *skip_until) is empty).
139  //
140  // Caveats:
141  // - The keys are skipped even if there are snapshots containing them,
142  // i.e. values removed by kRemoveAndSkipUntil can disappear from a
143  // snapshot - beware if you're using TransactionDB or
144  // DB::GetSnapshot().
145  // - If value for a key was overwritten or merged into (multiple Put()s
146  // or Merge()s), and `CompactionFilter` skips this key with
147  // kRemoveAndSkipUntil, it's possible that it will remove only
148  // the new value, exposing the old value that was supposed to be
149  // overwritten.
150  // - Doesn't work with PlainTableFactory in prefix mode.
151  // - If you use kRemoveAndSkipUntil for table files created by
152  // compaction, consider also reducing compaction_readahead_size
153  // option.
154  //
155  // Should never return kUndetermined.
156  // Note: If you are using a TransactionDB, it is not recommended to filter
157  // out or modify merge operands (ValueType::kMergeOperand).
158  // If a merge operation is filtered out, TransactionDB may not realize there
159  // is a write conflict and may allow a Transaction to Commit that should have
160  // failed. Instead, it is better to implement any Merge filtering inside the
161  // MergeOperator.
162  virtual Decision FilterV2(int level, const Slice& key, ValueType value_type,
163  const Slice& existing_value, std::string* new_value,
164  std::string* /*skip_until*/) const {
165  switch (value_type) {
166  case ValueType::kValue: {
167  bool value_changed = false;
168  bool rv = Filter(level, key, existing_value, new_value, &value_changed);
169  if (rv) {
170  return Decision::kRemove;
171  }
172  return value_changed ? Decision::kChangeValue : Decision::kKeep;
173  }
174  case ValueType::kMergeOperand: {
175  bool rv = FilterMergeOperand(level, key, existing_value);
176  return rv ? Decision::kRemove : Decision::kKeep;
177  }
178  case ValueType::kBlobIndex:
179  return Decision::kKeep;
180  }
181  assert(false);
182  return Decision::kKeep;
183  }

218 // Each thread of work involving creating table files will create a new

230  // Returns whether a thread creating table files for the specified `reason`
231  // should invoke `CreateCompactionFilter()` and pass KVs through the returned
232  // filter.
233  virtual bool ShouldFilterTableFileCreation(
234  TableFileCreationReason reason) const {
235  // For backward compatibility, default implementation only applies
236  // `CompactionFilter` to files generated by compaction.
237  return reason == TableFileCreationReason::kCompaction;
