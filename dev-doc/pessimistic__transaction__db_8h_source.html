72  inline Status WriteWithConcurrencyControl(const WriteOptions& opts,
73  WriteBatch* updates) {
74  // Need to lock all keys in this batch to prevent write conflicts with
75  // concurrent transactions.
76  Transaction* txn = BeginInternalTransaction(opts);
77  txn->DisableIndexing();
78 
79  auto txn_impl = static_cast_with_check<PessimisticTransaction>(txn);
80 
81  // Since commitBatch sorts the keys before locking, concurrent Write()
82  // operations will not cause a deadlock.
83  // In order to avoid a deadlock with a concurrent Transaction, Transactions
84  // should use a lock timeout.
85  Status s = txn_impl->CommitBatch(updates);
86 
87  delete txn;
88 
89  return s;
90  }
