20 // A very simple random number generator. Not especially good at
21 // generating truly random bits, but good enough for our needs in this
22 // package.
23 class Random {
24  private:
25  enum : uint32_t {
26  M = 2147483647L // 2^31-1
27  };
28  enum : uint64_t {
29  A = 16807 // bits 14, 8, 7, 5, 2, 1, 0
30  };
31 
32  uint32_t seed_;
33 
34  static uint32_t GoodSeed(uint32_t s) { return (s & M) != 0 ? (s & M) : 1; }
35 
36  public:
37  // This is the largest value that can be returned from Next()
38  enum : uint32_t { kMaxNext = M };
39 
40  explicit Random(uint32_t s) : seed_(GoodSeed(s)) {}
41 
42  void Reset(uint32_t s) { seed_ = GoodSeed(s); }
43 
44  uint32_t Next() {
45  // We are computing
46  // seed_ = (seed_ * A) % M, where M = 2^31-1
47  //
48  // seed_ must not be zero or M, or else all subsequent computed values
49  // will be zero or M respectively. For all other values, seed_ will end
50  // up cycling through every number in [1,M-1]
51  uint64_t product = seed_ * A;
52 
53  // Compute (product % M) using the fact that ((x << 31) % M) == x.
54  seed_ = static_cast<uint32_t>((product >> 31) + (product & M));
55  // The first reduction may overflow by 1 bit, so we may need to
56  // repeat. mod == M is not possible; using > allows the faster
57  // sign-bit-based test.
58  if (seed_ > M) {
59  seed_ -= M;
60  }
61  return seed_;
62  }
63 
64  uint64_t Next64() { return (uint64_t{Next()} << 32) | Next(); }
65 
66  // Returns a uniformly distributed value in the range [0..n-1]
67  // REQUIRES: n > 0
68  uint32_t Uniform(int n) { return Next() % n; }
69 
70  // Randomly returns true ~"1/n" of the time, and false otherwise.
71  // REQUIRES: n > 0
72  bool OneIn(int n) { return Uniform(n) == 0; }
73 
74  // "Optional" one-in-n, where 0 or negative always returns false
75  // (may or may not consume a random value)
76  bool OneInOpt(int n) { return n > 0 && OneIn(n); }
77 
78  // Returns random bool that is true for the given percentage of
79  // calls on average. Zero or less is always false and 100 or more
80  // is always true (may or may not consume a random value)
81  bool PercentTrue(int percentage) {
82  return static_cast<int>(Uniform(100)) < percentage;
83  }
84 
85  // Skewed: pick "base" uniformly from range [0,max_log] and then
86  // return "base" random bits. The effect is to pick a number in the
87  // range [0,2^max_log-1] with exponential bias towards smaller numbers.
88  uint32_t Skewed(int max_log) {
89  return Uniform(1 << Uniform(max_log + 1));
90  }
91 
92  // Returns a random string of length "len"
93  std::string RandomString(int len);
94 
95  // Generates a random string of len bytes using human-readable characters
96  std::string HumanReadableString(int len);
97 
98  // Generates a random binary data
99  std::string RandomBinaryString(int len);
100 
101  // Returns a Random instance for use by the current thread without
102  // additional locking
103  static Random* GetTLSInstance();
104 };

106 // A good 32-bit random number generator based on std::mt19937.
107 // This exists in part to avoid compiler variance in warning about coercing
108 // uint_fast32_t from mt19937 to uint32_t.
109 class Random32 {
110  private:
111  std::mt19937 generator_;
112 
113  public:
114  explicit Random32(uint32_t s) : generator_(s) {}
115 
116  // Generates the next random number
117  uint32_t Next() { return static_cast<uint32_t>(generator_()); }
118 
119  // Returns a uniformly distributed value in the range [0..n-1]
120  // REQUIRES: n > 0
121  uint32_t Uniform(uint32_t n) {
122  return static_cast<uint32_t>(
123  std::uniform_int_distribution<std::mt19937::result_type>(
124  0, n - 1)(generator_));
125  }
126 
127  // Returns an *almost* uniformly distributed value in the range [0..n-1].
128  // Much faster than Uniform().
129  // REQUIRES: n > 0
130  uint32_t Uniformish(uint32_t n) {
131  // fastrange (without the header)
132  return static_cast<uint32_t>((uint64_t(generator_()) * uint64_t(n)) >> 32);
133  }
134 
135  // Randomly returns true ~"1/n" of the time, and false otherwise.
136  // REQUIRES: n > 0
137  bool OneIn(uint32_t n) { return Uniform(n) == 0; }
138 
139  // Skewed: pick "base" uniformly from range [0,max_log] and then
140  // return "base" random bits. The effect is to pick a number in the
141  // range [0,2^max_log-1] with exponential bias towards smaller numbers.
142  uint32_t Skewed(int max_log) {
143  return Uniform(uint32_t{1} << Uniform(max_log + 1));
144  }
145 
146  // Reset the seed of the generator to the given value
147  void Seed(uint32_t new_seed) { generator_.seed(new_seed); }
148 };
